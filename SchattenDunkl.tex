\begin{theorem}\label{schatten}
Suppose $N\geq 1$, {\color{red}$\max\{2,N\}<p<\infty$} and $b\in  L^1_{{\rm loc}}(\mathbb R^{N},dw)$. For any $j\in\{1,2,...,N\}$, we have  $[b,R_{\D,j}]\in S^p$

\subsection{Harmonic analysis in the Dunkl setting}
Let $\mathbb{R}^N$ be the Euclidean space equipped with the usual scalar product $\langle \cdot,\,\cdot\rangle $ and the induced norm $\|\cdot\|$. For a nonzero vector $v\in \mathbb{R}^N$, the reflection $\sigma_v$ with respect to the hyperplane $v^{\perp}$ orthogonal to $v$ is defined by
\begin{align*}
	\sigma_v x:= x-2\frac{\langle x,v\rangle}{\|v\|^2}v.
\end{align*}
A finite set $R\subset \mathbb{R}^N\backslash \{0\}$ is called a root system if $\sigma_v(R)=R$ for every $v\in R$. A root system $R$ is called a normalized reduced root system if  $\|v\|^2=2$ for every $v\in R$.
The finite group $G$ generated by the reflections $\{\sigma_v\}_{v\in R}$ is called the Weyl group (reflection group) of the root system $R$.
Let $k: R \to [0,\infty)$ be a multiplicity function, which is a $G$-invariant function  fixed throughout the paper. We will denote by $\mathcal O(x)=\{\sigma(x): \sigma \in G\}$ the $G$-orbit of a point $x\in\mathbb{R}^N$, and $\mathcal O(E)=\bigcup_{x \in E} \mathcal O(x)$ the $G$-orbit of $E\subset \mathbb{R}^N$.

Define the Dunkl measure $dw$ on $\mathbb{R}^N$ by
$$
dw(x) :=\prod_{v \in R} |\langle x,v\rangle|^{k(v)} dx,
$$
where $dx$ stands for the Lebesgue measure on $\mathbb{R}^N$. Let $B(x,r):=\{y\in \mathbb{R}^N:\|x-y\|< r\}$ be the Euclidean ball centred at $x$ with radius $r>0$. Let $\N:=N+\sum_{v\in R}k(v)$. Note that
$$w(B(tx,tr))=t^{\N}w(B(x,r)),\ {\rm for}\ x\in\mathbb{R}^N,\ t,r>0.$$
Furthermore,
\begin{align}\label{twosideinequ}
  w(B(x,r))\sim r^N \prod_{v \in R}(|\langle x, v \rangle |+r)^{k(v)},
\end{align}
where the implicit constant only depends on $N,R$ and $k$.  It follows from \eqref{twosideinequ} that there exists a constant $C\geq 1$ such that for every $x\in \mathbb{R}^N$,
\begin{align}\label{doublinginequality}
C^{-1}\left(\frac{r_2}{r_1}\right)^N\leq \frac{w(B(x,r_2))}{w(B(x,r_1))}\leq C\left(\frac{r_2}{r_1}\right)^\N,\ {\rm for}\ 0<r_1<r_2,
\end{align}
so the numbers $\N$ and $N$ are called the upper and lower homogeneous dimensions, respectively.

The Dunkl operators $T_\xi$ are defined by
$$T_\xi f(x):=\partial_\xi f(x)+\sum_{v\in R}\frac{k(v)}{2}\langle v,\xi\rangle\frac{f(x)-f(\sigma_v(x))}{\langle v,x\rangle},$$
where $\partial_\xi f$ denotes the directional derivative of $f$ in the direction $\xi$. Furthermore, the Dunkl Laplacian is defined by $$\Delta_{\D}f(x):=\sum_{j=1}^N T_{e_j}^2f(x)=\Delta_{\mathbb{R}^N}f(x)+\sum_{v\in R}k(v)\delta_v f(x),$$
where $\Delta_{\mathbb{R}^N}$ is the Laplacian on $\mathbb{R}^N$ and $\delta_v$ is defined by
$$\delta_v f(x):=\frac{\partial_v f(x)}{\langle v,x\rangle}-\frac{f(x)-f(\sigma_v(x))}{\langle v,x\rangle^2}.$$
In this article, we consider the Dunkl-Riesz transforms defined by $R_{\D,j}:=T_{e_j}(-\Delta_{\D})^{-1/2}$, $j=1,2,\ldots,N$.


\begin{definition}{\rm
The Weyl chambers are defined as the closures of connected components of the set
  \begin{align}
    \left\{x\in\mathbb{R}^N:\langle x,v\rangle \neq 0\ \ { for} \ { all} \ v\in R\right\}.\end{align}}
\end{definition}
Define $
	d(x,y):=\min_{\sigma\in G}\|x-\sigma(y)\|
$
as the distance between two $G$-orbits $\mathcal O(x)$ and $\mathcal O(y)$. Recall from \cite[Chapter VII, proof of Theorem 2.12]{MR514561} that
\begin{align}\label{distanceequa}
d(x,y)=\|x-y\|\ {\rm if}\ {\rm and}\ {\rm only}\ {\rm if}\ x,y\in\Gamma
\end{align}
for some closed Weyl chamber $\Gamma$.

It is clear that $\mathcal O(B(x,r)) = \{y\in \mathbb{R}^N: d(x,y)<r\}$. Moreover,
\[
w(B(x,r)) \leq w(\mathcal O(B(x,r))) \leq |G|w(B(x,r))
\]
for all $x \in \mathbb{R}^N$ and $r>0$. We also denote $B^d(x,r):= \{y\in \mathbb{R}^N: d(x,y)<r\}$, i.e., $B^d(x,r)=\mathcal O(B(x,r))$.

Denote by $K_{\D,j}(x,y)$ the integral kernel of $R_{\D,j}$. We recall the kernel estimates for the Riesz transforms in the Dunkl setting proved in \cite{MR4733961}. Although these estimates may not be sharp (see the comment following Theorem~2.1 in \cite[p.~10]{MR4733961}), they suffice for our purposes.
\begin{lemma}\label{kernelestimate}
There exists a constant $C$ such that for $j\in\{1,2,\ldots,N\}$ and every $x,y$ with $d(x,y)\neq 0$,
\begin{align*}
|K_{\D,j}(x,y)|\leq C\frac{d(x,y)}{\|x-y\|}\frac{1}{w(B(x,d(x,y)))}.
\end{align*}
\end{lemma}
\begin{proof}
The proof is given in \cite[Theorem 1.1]{MR4733961}.
\end{proof}

\begin{lemma}\label{nondegenerate}
For $j\in\{1,2,\ldots,N\}$, and for every ball $B=B(x_0,r)\subset \mathbb{R}^N$, there is another ball $\widehat{B}=B(y_0,r)$ with $y_0=x_0+5re_j$ such that for every $(x,y)\in B\times \widehat{B}$, $K_{\D,j}(x,y)$ does not change sign and satisfies
\begin{align*}
|K_{\D,j}(x,y)|\geq\frac{C}{w(B(x_0,r))}.
\end{align*}
\end{lemma}
\begin{proof}
The lower bound is given in \cite[Theorem 1.2]{MR4733961}, while the fact that $K_{\D,j}(x,y)$ does not change sign can be seen in the proof of \cite[Theorem 1.2]{MR4733961}. Indeed, with the choice $y_0:=x_0+5re_j$, we have $y_j-x_j\geq 3r$ for all $(x,y)\in B\times \widehat{B}$. Together with the fact that the heat kernel $h_t(x,y)$ associated with the Dunkl Laplacian is  positive, the representation
$$K_{\D,j}(x,y)=-\frac{C}{2}\int_0^\infty \frac{y_j-x_j}{t}h_t(x,y)\frac{dt}{\sqrt{t}},$$
implies that $K_{\D,j}(x,y)<0$ (and thus, does not change sign) on $ B\times \widehat{B}$.
\end{proof}

\begin{definition}\label{Defsystem}{\rm
Let {\color{red}$X\subseteq \mathbb{R}^N$ be a nonempty open set}. A countable family
$
    \mathscr{D}
    := \bigcup_{k\in\mathbb{Z}}\mathscr{D}_{k}, \
    \mathscr{D}_{k}
    :=\{Q^k_{\alpha}\colon \alpha\in \mathscr{A}_k\},
$
of Borel sets $Q^k_{\alpha}\subseteq X$ is called \textit{a
system of dyadic cubes over  $X$}  if it has the following properties:

\begin{enumerate}
  \item[(i)] For each $k\in\mathbb{Z}$, we have $$    X
    = \bigcup_{\alpha\in \mathscr{A}_k} Q^k_{\alpha};$$\smallskip

  \item[(ii)] $\text{If }\ell\geq k\text{, then either }
        Q^{\ell}_{\beta}\subseteq Q^k_{\alpha}\text{ or }
        Q^k_{\alpha}\cap Q^{\ell}_{\beta}=\varnothing$;\smallskip

  \item[(iii)] $
    \text{For each }(k,\alpha)\text{ and each } \ell\leq k,
    \text{ there exists a unique } \beta
    \text{ such that }Q^{k}_{\alpha}\subseteq Q^\ell_{\beta};
$

\smallskip

\item[(iv)] For each $(k,\alpha)$ there exist $2^N$ indices $\beta$ such that
    $$ Q^{k+1}_{\beta}\subseteq Q^k_{\alpha},\ {\rm and}\
        Q^k_{\alpha} =\bigcup_{{\substack{Q\in\mathscr{D}_{k+1}\\ Q\subseteq Q^k_{\alpha}}}}Q;$$

\smallskip

\item[(v)] For each $(k,\alpha)$, one has $$B_X(x^k_{\alpha},A_1 2^{-k})
    \subseteq Q^k_{\alpha}\subseteq B_X(x^k_{\alpha},A_2 2^{-k})
    =: B(Q^k_{\alpha})$$
    for some positive constants $A_1,A_2$ depending only on the geometry of $X$, {\color{red}where we set $B_X(x,r):=B(x,r)\cap X$ for all $x\in X$ and $r>0$.}

\smallskip

\item [(vi)]  If $\ell\geq k$ and
   $Q^{\ell}_{\beta}\subseteq Q^k_{\alpha}$, then
   $$B(Q^{\ell}_{\beta})\subseteq B(Q^k_{\alpha}).$$
\end{enumerate}



   }
\end{definition}

Each $Q^k_{\alpha}$ is called a \textit{dyadic cube of generation} $k$, with center $x^k_{\alpha}\in Q^k_{\alpha}$ and side-length $\ell(Q^k_{\alpha})=2^{-k}$.
\begin{nota}
For every dyadic cube $Q\in\mathscr{D}$, we denote by $c_Q$ and $\ell(Q)$ its center and side-length, respectively.
\end{nota}

\begin{nota}\label{standardsystem}
Let $$\mathbb{D}^0:=\bigcup_{k\in\mathbb{Z}}\mathbb{D}_k^0$$ be the standard system of dyadic cubes over $\mathbb{R}^N$.
Here each dyadic cube is a product of half-open intervals
\[
[m_1 2^{-k}, (m_1+1)2^{-k}) \times \cdots \times [m_N 2^{-k}, (m_N+1)2^{-k}),
\]
with $(m_1,\ldots,m_N)\in \mathbb{Z}^N$. Moreover, we let $$\mathbb{D}_{\Gamma}:=\bigcup_{k\in\mathbb{Z}}\mathbb{D}_{\Gamma,k}$$ be the system of dyadic cubes over a fixed chamber $\Gamma$, whose  construction will be given in Corollary \ref{Construction over gamma}.
\end{nota}

\begin{lemma}\label{Nsystems}
There exist $2^N$ systems of dyadic cubes $\mathbb{D}^1,\mathbb{D}^2,\ldots,\mathbb{D}^{2^N}$ such that for any ball $B=B(x_B,r_B)\subset \mathbb{R}^N$ with $\frac{2^{-k-2}}{3}\leq r_B< \frac{2^{-k-1}}{3}$ for some $k\in\mathbb{Z}$,  there exists a cube ${\displaystyle Q\in\bigcup_{\nu=1}^{2^N}\mathbb{D}^\nu_k}$ such that
$$B\subset Q\subset 12\sqrt{N} B.$$
\end{lemma}

\begin{proof}
{\bf $\bullet$ Step 1.}  {\it Recall the one-dimensional setting given in \cite[Section 2]{MR3420475}.}

Let $\mathbb{D}^{(0)}(\mathbb{R})$ be the standard system of dyadic intervals over $\mathbb{R}$ (see Notation \ref{standardsystem}).
Let $\delta=1/3$ and define
$$
   \mathbb D^{(\delta)}(\mathbb R)=\bigcup_{k\in\mathbb Z} \mathbb D^{(\delta)}_k(\mathbb R),
$$
where for $k\geq0$,
$$
    \mathbb D^{(\delta)}_k(\mathbb R)=\{I+\delta \mid I\in \mathbb D_k^{(0)}(\mathbb R)\},
$$
while for $k<0$ and $k$ even,
$$
    \mathbb D^{(\delta)}_k(\mathbb R)
    =\Big\{\Big[{m\over 2^k}+\delta+\sum_{j=(k/2)+1}^0 2^{-2j},\ {m+1\over 2^k}
    +\delta+\sum_{j=(k/2)+1}^0 2^{-2j}\Big)\ \Big|\ m\in
    \mathbb Z\Big\}.
$$
These choices together with the nestedness property property completely
determine the collections $\mathbb D^{(\delta)}_k(\mathbb R)$ for $k<0$, $k$ odd.

Fix an interval $J\subset\mathbb R$ with $2^{-k-1}/3\le |J|<2^{-k}/3$ for some $k\in\mathbb Z$. Now we set
$A_k=\{ m\cdot 2^{-k} \mid m\in\mathbb Z \}$ for every fixed $k$ and
\begin{enumerate}
    \item[(1)]$A_k^{(\delta)}=\{ \delta+ m\cdot 2^{-k} \mid m\in\mathbb Z
        \}$ for $k\geq 0$,

    \item[(2)]$A_k^{(\delta)}=\{ \delta+
        \sum_{j=(k/2)+1}^{0}2^{-2j}+ m\cdot 2^{-k} \mid m\in\mathbb Z
        \}$ for $k<0$, $k$ even, and

    \item[(3)]$A_k^{(\delta)}=\{ \delta+
        \sum_{j=(k-1)/2+1}^{0}2^{-2j}+ m\cdot 2^{-k} \mid
        m\in\mathbb Z \}$ for $k<0$, $k$ odd.
\end{enumerate}
Note that for any two different points $a,b\in A_k\cup
A_k^{(\delta)}$, we have $|a-b|\geq 2^{-k}/3>|J|$. Thus,
there is at most one element of $A_k\cup A_k^{(\delta)}$ contained in
$J$. Hence, $J\cap A_k=\emptyset $ or $J\cap
A_k^{(\delta)}=\emptyset $. Therefore, $J$ must be contained in
some dyadic interval $I\in \mathbb D_k^{(0)}$ or $I\in \mathbb D^{(\delta)}_k$ and
$|I|=2^{-k}\leq 6 |J|$.

{\bf $\bullet$ Step 2.} {\it We extend the construction to $\mathbb{R}^N$ ($N\geq 1$).}

For $\vartheta=(\vartheta_1,\vartheta_2,\ldots,\vartheta_N)\in\{0,\delta\}^N$, define
$$\mathbb{D}^{(\vartheta)}:=\bigcup_{k\in\mathbb{Z}}\mathbb{D}_k^{(\vartheta)},\quad \mathbb{D}_k^{(\vartheta)}:=\left\{\prod_{j=1}^N I_j: I_j\in \mathbb{D}_k^{(\vartheta_j)}\right\}.$$
Applying Step 1 to $\pi_i(B)$, where $\pi_i(B)$ denotes the $i$-th coordinate projection, we obtain $I_i\in\mathbb{D}_k^{(0)}$ or $I_i\in\mathbb{D}_k^{(\delta)}$ with $\pi_i(B)\subset I_i$ and $|I_i|=2^{-k}\leq 12 r_B$. Let $\vartheta_i\in\{0,\delta\}$
be the corresponding shift so that $I_i\in\mathbb D_k^{(\vartheta_i)}$, set $\vartheta:=(\vartheta_1,\ldots,\vartheta_N)$, and let $Q=\prod_{j=1}^NI_j\in\mathbb{D}_k^{(\vartheta)}$. It follows that $B\subset Q$ and $\ell(Q)=2^{-k}\leq 12r_B$. Hence, for every $y\in Q$,
$$\|y-x_B\|\leq \sqrt{N}\ell(Q)\leq 12\sqrt{N}r_B.$$
This implies that $Q\subset B(x_B,12\sqrt{N}r_B)$. Enumerating $\big\{\mathbb D^{(\vartheta)}:\vartheta\in\{0,\delta\}^N\big\}$ as
$\mathbb D^1,\ldots,\mathbb D^{2^N}$, we complete the proof of Lemma \ref{Nsystems}.
\end{proof}

\subsection{Haar Basis}\label{HaarSection}

{\color{red}{\color{red}[Zhijie will add more background later]}}

Given a system of dyadic cubes $\mathscr{D}$ over $\mathbb{R}^N$, we now recall the standard way to construct an orthonormal Haar system corresponding to $\mathscr{D}$. For each $k\in\mathbb{Z}$ and $Q\in\mathscr{D}_k$, denote by $Ch(Q):=\{R\in\mathscr{D}_{k+1}:R\subseteq Q\}$. Moreover,  we let
$$h_Q^0:=w(Q)^{-1/2}\chi_Q.$$
Fix an enumeration $\{Q_i\}_{i=1}^{2^N}$ of $Ch(Q)$. Let
$$\hat{Q}_i:=\bigcup_{j=i}^{2^N}Q_j$$
and
$$h_Q^i:=\sqrt{\frac{w(Q_i)w(\hat{Q}_{i+1})}{w(\hat{Q}_i)}}\Bigg(\frac{\chi_{Q_i}}{w(Q_i)}-\frac{\chi_{\hat{Q}_{i+1}}}{w(\hat{Q}_{i+1})}\Bigg),\quad i=1,2,\ldots,2^N-1.$$


\begin{lemma}\label{Haarexpansion}
For any $1<p<\infty$ and $f\in
    L^p(\mathbb{R}^N,dw)$, we have
    \[
        f(x)
        =  \sum_{Q\in\mathscr{D}}\sum_{i=1}^{2^N-1}
            \langle f,h^{i}_{Q}\rangle_{L^2(\mathbb{R}^N,dw)} h^{i}_{Q}(x), %\quad \text{if } \mu(X)=\infty,
    \]
    where the sum converges both in the

 \end{lemma}

\subsection{Nearly weakly  orthonormal sequences}
Rochberg and Semmes \cite{RS} introduced the notion of \emph{nearly weakly  orthonormal (NWO)} sequences of functions. This notion has since proved highly effective in characterizing Schatten class membership of  singular integral commutators in a variety of contexts (see \cite{FLLXarxiv,HKarxiv,MR4756021,MR4552581,MR4873796,LLW,RS,Zhangwei1,Zhangwei2}). The properties of NWO sequences were originally investigated in the Euclidean setting (equipped with Lebesgue measure), but the extension to spaces of homogeneous type is straightforward once a system of dyadic cubes is available in this general framework (see \cite[Section 13]{HKarxiv} for a complete proof).  For our purposes, we adapt  the extension to the Dunkl setting $(\mathbb{R}^N,\|\cdot\|,w)$ rather than recalling the general  definition
of NWO sequences and the general extension to spaces of homogeneous type.

{\it Let $\mathscr{D}$ be a system of dyadic cubes. A sequence of functions $\{e_Q\}_{Q\in\mathscr{D}}$ is an NWO sequence if
\begin{enumerate}
  \item $\supp e_Q\subseteq cQ$ for some positive constant $c$ independent of $Q$
  \item $\|e_Q\|_{L^s(\mathbb{R}^{N},dw)}\leq w(Q)^{1/s-1/2}$ for some $s>2$.
\end{enumerate}
}
\begin{lemma}\label{NWOLEMMAA}
Let $1<p<\infty$, and let $\mathscr{D}$ be a
system of dyadic cubes over $\mathbb{R}^N$. For any compact operator $ T $ on $ L ^2 (\mathbb R ^{N},dw)$, we have
\begin{equation*}
\Bigl(
\sum_{Q\in \mathscr{D}} |\langle T e_Q, f_Q \rangle_{L^2(\mathbb{R}^N,dw)} | ^{p}
\Bigr) ^{1/p} \leq C_{N,p,w} \| T \| _{S ^{p}},
\end{equation*}
where $\{e_Q\}_{Q\in\mathscr{D}}$ and $\{f_Q\}_{Q\in\mathscr{D}}$ are NWO sequences.
\end{lemma}
\begin{proof}
This property can be found in  \cite[(1.10)]{RS} (see also \cite[Section 7]{HKarxiv}).
\end{proof}

\begin{lemma}\label{NWOLEMMAA2}
Let $0<p<\infty$, and let $\mathscr{D}$ be a
system of dyadic cubes over $\mathbb{R}^N$. Assume that
$$A=\sum_{Q\in\mathscr{D}}\lambda_Q\langle \cdot,\,e_Q\rangle_{L^2(\mathbb{R}^N,dw)} f_Q,$$
where $\{e_Q\}_{Q\in\mathscr{D}}$ and $\{f_Q\}_{Q\in\mathscr{D}}$ are NWO sequences and $\{\lambda_Q\}_{Q\in\mathscr{D}}$ is a sequence of scalars. We have
\begin{align}\label{nwose}
\|A\|_{S^{p}}\leq C_{N,p,w} \|\{\lambda_Q\}_Q\|_{\ell^{p}}.
\end{align}
\end{lemma}
\begin{proof}
This property can be found in  \cite[(1.9)]{RS} (see also \cite[Section 7]{HKarxiv}).
\end{proof}

  \item Let

  \Omega:=\mathbb{R}_+^{N_{0}}\times \mathbb{R}^{N-N_{0}};

\begin{definition}{\rm \cite[Definition ~2.7]{HeckmanCoxeterGroups}}\label{def:weyl_chamber_positive_roots} A connected component of \begin{align}
    \left\{x\in\mathbb{R}^N:\langle x,v\rangle \neq 0\ \ { for} \ { all} \ v\in R\right\}
  \end{align} is called an open
Weyl chamber. Its closure is called a Weyl chamber.  Define the corresponding sets of \emph{positive} and \emph{negative} roots by
\[
R_{+}:=\{\alpha\in R:\ \langle x,\alpha\rangle>0\ \text{for all }x\in \Gamma_{+}\},
\qquad
R_{-}:=-R_{+}.
\]
\end{definition}

\begin{definition}{\rm \cite[Definition~2.9]{HeckmanCoxeterGroups}}\label{def:simple_roots}
A root $\alpha\in R_{+}$
is called simple in $R_{+}$ if it cannot be written as
\[
\alpha=\lambda_{1}\alpha_{1}+\lambda_{2}\alpha_{2}
\quad\text{with }\lambda_{1},\lambda_{2}\ge 1\ \text{and}\ \alpha_{1},\alpha_{2}\in R_{+}.
\]
The set $\Delta\subset R_{+}$ of all simple roots is called the
simple system.
\end{definition}

\begin{lemma}\label{lem:simple_system_associated_to_chamber}
With the notation given in Definition \ref{def:simple_roots},  $\Delta$ is a basis of $W$. If we denote by
$
\Delta=\{\alpha_{1},\alpha_{2},\ldots,\alpha_{N_{0}}\},
$
then the open and closed chambers can be written as
\begin{align}
\Gamma_{+}
&=\Bigl\{x\in\mathbb{R}^{N}:\ \langle x,\alpha_{i}\rangle>0,\ i=1,\ldots,N_{0}\Bigr\},
\label{eq:open_chamber_by_simple_roots}\\
\Gamma
&=\Bigl\{x\in\mathbb{R}^{N}:\ \langle x,\alpha_{i}\rangle\ge 0,\ i=1,\ldots,N_{0}\Bigr\},
\label{eq:closed_chamber_by_simple_roots}
\end{align}
and every root $\alpha\in R$ admits a unique expansion
\[
\alpha=\sum_{i=1}^{N_{0}}c_{i}\alpha_{i}
\]
such that either $c_{i}\ge 0$ for all $i$ (when $\alpha\in R_{+}$), or $c_{i}\le 0$ for all $i$
(when $\alpha\in R_{-}$). In particular, the simple system $\Delta$ is uniquely determined by $\Gamma_{+}$
(equivalently, by $\Gamma$).
\end{lemma}

\begin{proof}
%We regard $R$ as a normalized root system in the Euclidean space $W$ endowed with the
%restricted inner product. Since $\sigma_{\alpha}(\alpha)=-\alpha$ and $\sigma_{\alpha}(R)=R$
%for every $\alpha\in R$, we have $R=-R$.

Let $\Delta=\{\alpha_{1},\ldots,\alpha_{n}\}\subset R_{+}$ be the set of simple roots in $R_{+}$. By \cite[Proposition~2.10]{HeckmanCoxeterGroups}, every $\alpha\in R_{+}$
is a nonnegative linear combination of $\alpha_{1},\ldots,\alpha_{n}$. Since $R=R_{+}\cup R_{-}$ (this follows from the fact that for each $\alpha\in R$, the continuous function
$x\mapsto\langle x,\alpha\rangle$ does not vanish on the connected set $\Gamma_{+}$) and $R_{-}=-R_{+}$, it follows that
$R\subset\operatorname{span}(\Delta)$ and hence,
\[
W=\operatorname{span}(R)\subset\operatorname{span}(\Delta).
\]
The reverse inclusion $\operatorname{span}(\Delta)\subset W$ is immediate from $\Delta\subset R\subset W$,
so $\operatorname{span}(\Delta)=W$. On the other hand, it follows from \cite[Theorem~2.16]{HeckmanCoxeterGroups} that the simple roots are linearly
independent. Consequently, $\Delta$ is a basis
of $W$, and therefore $n=\dim W=N_{0}$. Moreover, every $\alpha\in R$ has a unique expansion
$\alpha=\sum_{i=1}^{N_{0}}c_{i}\alpha_{i}$. If $\alpha\in R_{+}$, the coefficients can be chosen
nonnegative by \cite[Proposition~2.10]{HeckmanCoxeterGroups}. If $\alpha\in R_{-}=-R_{+}$, then
$-\alpha\in R_{+}$ has an expansion with all coefficients nonnegative, and multiplying by $-1$ gives
an expansion of $\alpha$ with all coefficients nonpositive.
The descriptions \eqref{eq:open_chamber_by_simple_roots} and \eqref{eq:closed_chamber_by_simple_roots} follow from \cite[Corollary~2.11]{HeckmanCoxeterGroups}.


To see that $\Delta$ is uniquely determined by $\Gamma_+$, note that $\Gamma_+$ determines $R_{+}$
by Definition~\ref{def:weyl_chamber_positive_roots}, and the set of simple roots in $R_{+}$ is
characterized intrinsically by Definition~\ref{def:simple_roots}. Finally, since $\Gamma_+$ is the (unique) interior component of $\Gamma$, specifying $\Gamma$ determines $\Gamma_+$ and vice versa. This completes the proof of Lemma \ref{lem:simple_system_associated_to_chamber}.
\end{proof}
\begin{nota}
With the notation given in Lemma \ref{lem:simple_system_associated_to_chamber}, denote by
 $$\Delta=\{\alpha_{1},\ldots,\alpha_{N_{0}}\}$$ the simple system associated with $\Gamma$.
\end{nota}

\begin{lemma}\label{chambergeometry}
Each Weyl chamber $\Gamma$ admits the decomposition
\[
\Gamma = K(\Gamma)\times W^\perp,
\qquad
K(\Gamma):=\{w\in W:\ \langle w,\alpha\rangle\ge 0\ \text{for all }\alpha\in\Delta\}.
\]
\end{lemma}

\begin{proof}
Write $x=w+z$ with $w\in W$ and $z\in W^{\perp}$. Since $\alpha_{i}\in W$, we have $\langle x,\alpha_{i}\rangle=\langle w,\alpha_{i}\rangle$ for all
$i=1,\ldots,N_0$. This, together with \eqref{eq:closed_chamber_by_simple_roots}, implies that $x\in \Gamma$ if and only if
$
w\in K(\Gamma).
$
This implies that
$
\Gamma
=
K(\Gamma)\times W^{\perp},
$
and thus completes the proof.
\end{proof}

\begin{coro}\label{coro:chamber-orthant}
Let $\Phi_{W^{\perp}}$ be any linear isomorphism from $W^\perp$ to $\mathbb{R}^{N-N_{0}}$, and let
\[
\Phi_W:W\longrightarrow\mathbb{R}^{N_{0}},
\qquad
\Phi_W(w):=\big(\langle w,\alpha_1\rangle,\dots,\langle w,\alpha_{N_{0}}\rangle\big),
\]
\[
\Phi:\mathbb{R}^N=W\oplus W^\perp\longrightarrow
\mathbb{R}^{N_{0}}\oplus\mathbb{R}^{N-N_{0}}\cong\mathbb{R}^N,
\qquad
\Phi(w+z):=\big(\Phi_W(w),\,\Phi_{W^\perp}(z)\big).
\]
Then $\Phi$ is a linear isomorphism and
\[
\Phi(\Gamma)=\mathbb{R}_+^{N_{0}}\times \mathbb{R}^{N-N_{0}}.
\]
\end{coro}

\begin{proof}
By Lemma~\ref{chambergeometry} we have $\Gamma=K(\Gamma)\times W^\perp$, where
\[
K(\Gamma)=\{w\in W:\ \langle w,\alpha_i\rangle\ge0,\ i=1,\dots,N_{0}\}.
\]
Since $\Delta$ is a basis of $W$, the Gram matrix
$G=(\langle \alpha_i,\alpha_j\rangle)_{i,j=1}^{N_{0}}$ is nonsingular. Hence,
$\Phi_W:W\longrightarrow\mathbb{R}^{N_{0}}$ is a linear isomorphism. Moreover,
for $w\in W$ we have $w\in K(\Gamma)$ if and only if $\Phi_{W}(w)\in\mathbb{R}_+^{N_{0}}$,
so
\[
\Phi_{W}\big(K(\Gamma)\big)=\mathbb{R}_+^{N_{0}}.
\]
By construction, $\Phi$ is a linear isomorphism (as a direct sum of isomorphisms),
and therefore
\[
\Phi(\Gamma)
=\Phi\big(K(\Gamma)\times W^\perp\big)
=\Phi_{W}\big(K(\Gamma)\big)\times \Phi_{W^{\perp}}\big(W^\perp\big)
=\mathbb{R}_+^{N_{0}}\times \mathbb{R}^{N-N_{0}},
\]
which proves the desired conclusion.
\end{proof}


\begin{coro}\label{Construction over gamma}
 Let $\Gamma$ be an arbitrary Weyl chamber. There exists a system of dyadic cubes over $\Gamma$.
\end{coro}\begin{proof}
Let $\Phi:\mathbb{R}^N\to\mathbb{R}^N$ be the linear isomorphism from Corollary~\ref{coro:chamber-orthant} so that
\[
\Phi(\Gamma)=\Omega:=\mathbb{R}_+^{N_0}\times \mathbb{R}^{N-N_0}.
\]

For each $k\in\mathbb{Z}$, define $\mathbb{D}^0_{\Omega,k}$ to be the family of half-open cubes
\[
Q=\prod_{i=1}^{N} I_i,
\qquad
I_i=\big[m_i2^{-k},(m_i+1)2^{-k}\big),
\]
where $m_i\in\mathbb{Z}_+$ for $1\le i\le N_0$ and $m_i\in\mathbb{Z}$ for $N_0<i\le N$. Set
\[
\mathbb{D}^0_\Omega:=\bigcup_{k\in\mathbb{Z}}\mathbb{D}^0_{\Omega,k}.
\]
It is standard that $\mathbb{D}^0_\Omega$ is a system of dyadic cubes over $\Omega$.

Next, for each $k\in\mathbb{Z}$, define
\[
\mathbb{D}_{\Gamma,k}:=\big\{\Phi^{-1}(Q): Q\in \mathbb{D}^0_{\Omega,k}\big\},
\qquad
\mathbb{D}_{\Gamma}:=\bigcup_{k\in\mathbb{Z}}\mathbb{D}_{\Gamma,k}.
\]
Since $\Phi:\mathbb{R}^N\to\mathbb{R}^N$ is a linear isomorphism, we see that $\mathbb{D}_{\Gamma}$ is a system of dyadic cubes over $\Gamma$.
\end{proof}




\begin{nota}
 Let $\Phi$ be the linear isomorphism given in Corollary~\ref{coro:chamber-orthant}. For every Borel set $E\subset\Omega$, define
\begin{align}\label{defmu}
  \mu(E):=w\big(\Phi^{-1}(E)\big).
\end{align}
\end{nota}
\begin{lemma}\label{mulemma}
We have
  \begin{equation}\label{pushforward-density}
d\mu(u,z)=\phi(u)\,du\,dz,\quad \phi(u)=|{\rm det}(\Phi)|^{-1}\prod_{v\in R_+}\Big(\sum_{i=1}^{N_{0}} a_{v,i}\,u_i\Big)^{2k(v)}.
\end{equation}
where for each $v\in R_+$, the coefficients $a_{v,i}\ge0$ are uniquely determined by the simple-root expansion
\(v=\sum_{i=1}^{N_0} a_{v,i}\alpha_i\) and satisfy $(a_{v,1},\dots,a_{v,N_0})\neq\mathbf 0$.
\end{lemma}
\begin{proof}
Since $\Phi(\Gamma)=\Omega$, we have $\Phi^{-1}(\Omega)=\Gamma$. Hence, for every nonnegative Borel function
$f$ on $\Omega$,
\[
\int_{\Omega} f(u,z)\,d\mu(u,z)
=\int_{\Gamma} f\big(\Phi(x)\big)\,dw(x).
\]
We make the linear change of variables $(u,z)=\Phi(x)$, so that $x=\Phi^{-1}(u,z)$. Since $\Phi$ is linear and invertible, the Jacobian is constant and
\[
dx = |{\rm det}(\Phi)|^{-1}\,du\,dz.
\]
It follows from the definition of $\Phi$ that
$
u_i=\langle x,\alpha_i\rangle$ for $i=1,\dots,N_0.
$
This implies that
\begin{align}\label{changevariable}
\langle x,v\rangle
=\Big\langle x,\sum_{i=1}^{N_0} a_{v,i}\alpha_i\Big\rangle
=\sum_{i=1}^{N_0} a_{v,i}\langle x,\alpha_i\rangle
=\sum_{i=1}^{N_0} a_{v,i}u_i.
\end{align}
Substituting these identities into the previous integral, we obtain
\begin{align*}
\int_{\Omega} f(u,z)\,d\mu(u,z)
&=\int_{\Gamma} f\big(\Phi(x)\big)\Big(\prod_{v\in R_+}|\langle x,v\rangle|^{2k(v)}\Big)\,dx \\
&=\int_{\Omega} f(u,z)\,
|{\rm det}(\Phi)|^{-1}\prod_{v\in R_+}\Big(\sum_{i=1}^{N_0} a_{v,i}u_i\Big)^{2k(v)}\,du\,dz.
\end{align*}
Therefore, \eqref{pushforward-density} holds. This completes the proof of Lemma~\ref{mulemma}.
\end{proof}


\begin{lemma}\label{lem:gallery-simple}
Let $\Gamma_1$ and $\Gamma_2$ be two Weyl chambers. Then there exist an integer $M\ge 0$ and
$\alpha_1,\dots,\alpha_M\in R$ such that, if we set $\Gamma^{(0)}:=\Gamma_2$ and
\[
\Gamma^{(k)}:=\sigma_{\alpha_k}\big(\Gamma^{(k-1)}\big),\qquad k=1,\dots,M,
\]
then the following hold:
\begin{enumerate}
  \item\label{geo1} $\alpha_k\in \Delta(\Gamma^{(k-1)})$ for each $k=1,\dots,M$;
  \item\label{geo2} $\Gamma^{(M)}=\Gamma_1$.
\end{enumerate}
\end{lemma}
\begin{proof}
If $\Gamma_1=\Gamma_2$, we take $M=0$ and there is nothing to prove. Assume $\Gamma_1\neq \Gamma_2$.
Since the Weyl group $G$ acts transitively on the set of Weyl chambers (see \cite[Theorem 2.12 (2)]{HeckmanCoxeterGroups}), there exists
$g\in G$ such that $\Gamma_1=g(\Gamma_2)$.

Let $\Delta(\Gamma_2)=\{\beta_1,\beta_2,\ldots,\beta_{N_0}\}$ be the simple system associated with $\Gamma_2$.
By \cite[Theorem 2.12(4)]{HeckmanCoxeterGroups}, the Weyl group $G$ is generated by the simple reflections $\sigma_{\beta_1},\sigma_{\beta_2},\ldots,\sigma_{\beta_{N_0}}$. Hence, there exist indices
$i_1,\ldots,i_M\in\{1,\ldots,N_0\}$ such that
\[
g=\sigma_{\beta_{i_1}}\sigma_{\beta_{i_2}}\cdots \sigma_{\beta_{i_M}}.
\]
Define
\[
w_0:=\mathrm{Id},\qquad w_k:=\sigma_{\beta_{i_1}}\cdots \sigma_{\beta_{i_k}}\quad (k=1,\ldots,M),
\]
and set
\[
\Gamma^{(k)}:=w_k(\Gamma_2)\qquad (k=0,1,\ldots,M).
\]
It follows that $\Gamma^{(0)}=\Gamma_2$ and $\Gamma^{(M)}=w_M(\Gamma_2)=g(\Gamma_2)=\Gamma_1$, which implies the statement \eqref{geo2}.

For each $k=1,\ldots,M$, define
\[
\alpha_k:=w_{k-1}(\beta_{i_k})\in R.
\]
The rest of the proof is divided into two steps.

{\bf $\bullet$ Step 1.} {\it Verify that $\Gamma^{(k)}=\sigma_{\alpha_k}(\Gamma^{(k-1)})$.}

Since each $w\in G$ is orthogonal, for any root $\beta\in R$ one has the conjugation identity
\[
w\,\sigma_{\beta}\,w^{-1}=\sigma_{w(\beta)},\qquad w\in G.
\]
Applying this with $w=w_{k-1}$ and $\beta=\beta_{i_k}$ yields
\[
\sigma_{\alpha_k}=\sigma_{w_{k-1}(\beta_{i_k})}=w_{k-1}\sigma_{\beta_{i_k}}w_{k-1}^{-1}.
\]
Therefore,
\[
\sigma_{\alpha_k}(\Gamma^{(k-1)})
= w_{k-1}\sigma_{\beta_{i_k}}w_{k-1}^{-1}\big(w_{k-1}(\Gamma_2)\big)
= w_{k-1}\sigma_{\beta_{i_k}}(\Gamma_2)
= w_k(\Gamma_2)
= \Gamma^{(k)}.
\]

{\bf $\bullet$ Step 2.} {\it Verify that $\alpha_k\in\Delta(\Gamma^{(k-1)})$.}

Recall that if $\Gamma$ is a Weyl chamber associated with simple system $\Delta(\Gamma)$, then
\[
\Gamma=\{x\in\mathbb R^N:\ \langle x,\gamma\rangle\ge 0\ \text{for all }\gamma\in\Delta(\Gamma)\}.
\]
Applying $w_{k-1}\in G$ to the above description for $\Gamma_2$ and using
$\langle w_{k-1}x,\,\beta\rangle=\langle x,\,w_{k-1}^{-1}\beta\rangle$, we obtain
\[
\Gamma^{(k-1)}=w_{k-1}(\Gamma_2)
=\Bigl\{x\in\mathbb R^N:\ \langle x,\,w_{k-1}(\beta_j)\rangle\ge 0\ \text{for }j=1,\ldots,N_0\Bigr\}.
\]
By the uniqueness of the simple system associated with a Weyl chamber
(see Lemma~\ref{lem:simple_system_associated_to_chamber}), we conclude that
\[
\Delta(\Gamma^{(k-1)})=\Delta\big(w_{k-1}(\Gamma_2)\big)=w_{k-1}\big(\Delta(\Gamma_2)\big)=\{w_{k-1}(\beta_1),\ldots,w_{k-1}(\beta_{N_0})\}.
\]
In particular, $\alpha_k=w_{k-1}(\beta_{i_k})\in\Delta(\Gamma^{(k-1)})$, proving the first statement \eqref{geo1}.

This completes the proof of Lemma \ref{lem:gallery-simple}.
\end{proof}


\section{Proof of upper bound}
\setcounter{equation}{0}

{\color{red}[Zhijie will add some Comment about the idea in this section later]}

Let $(X,\mu)$ be a measure space and let $L^p(L^{q,r})$ denote the mixed-norm space consisting of all $\mu$-measurable functions $f:X\times X \to \mathbb{C}$ such that
\[
\|f\|_{L^p(L^{q,r})} := \Big\|\, \|f(x,y)\|_{L^{q,r}_y(X,\mu)}\,\Big\|_{L^p_x(X,\mu)} < \infty.
\]
Next, we define
$$L^p(L^{q,r})^{\rm symm}:=L_x^{p}(L_y^{q,r})\cap L_y^p(L_x^{q,r}),$$
equipped with the norm
$$\|f\|_{L^p(L^{q,r})^{\rm symm}}:=\max\{\|f\|_{L_x^{p}(L_y^{q,r})},\|f\|_{L_y^p(L_x^{q,r})}\}.$$

\begin{lemma}\label{Russo}
Let $(X,\mu)$ be a measure space and let $2<p<\infty$. Suppose that $K:X\times X\rightarrow \mathbb{C}$ belongs to $L^p(L^{p',\infty})^{\rm symm}$. The integral operator $$T_Kf(x):=\int_X K(x,y)f(y)d\mu(y)$$
belongs to $S^{p,\infty}(L^2(X,d\mu))$. Moreover, we have
\begin{align}
  \|T_K\|_{S^{p,\infty}(L^2(X,d\mu))}\lesssim \|K\|_{L^p(L^{p',\infty})^{\rm symm}}.
\end{align}
\end{lemma}
\begin{proof}
The proof is given in \cite[Lemma 2]{JW} (see also \cite[Corollary 11.2]{HKarxiv}).
\end{proof}
In what follows, let $X=\mathbb{R}^N$ and $d\mu=dw$ for the underlying measure.
\begin{proposition}\label{upper bound}
    Let ${\color{red}p>\max\{2,N\}}$ and $b\in B^p_{\D}(\mathbb{R}^N)$. For any $j\in\{1,2,\ldots,N\}$, we have
    $$\|[b,R_{\D,j}]\|_{S^{p}(L^2(\mathbb{R}^N,dw))}\lesssim\|b\|_{B^p_{\D}(\mathbb{R}^N)}.$$
\end{proposition}
\begin{proof}
Our strategy is to adjust the argument in \cite[Proposition 11.8]{HKarxiv} to obtain the desired result.
For any $(w\otimes w)$-measurable function $F$ on $\mathbb{R}^{2N}$, we define a linear map $\mathcal{T}:F\rightarrow \mathcal{T}(F)$, where $\mathcal{T}(F)$ is an integral operator on $L^2(\mathbb{R}^N,dw)$ with kernel
$$\mathbb{R}^N\times\mathbb{R}^N\ni(x,y)\mapsto \frac{\|x-y\|}{d(x,y)}K_{\D,j}(x,y)F(x,y),\quad d(x,y)\neq 0.$$
Now we {\it claim} that $\mathcal{T}$ is a bounded linear map from $L^p(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y))$ to $S^{p,\infty}(L^2(\mathbb{R}^N,dw))$ for all $p>2$. To see this, we apply Lemma \ref{Russo} and then Lemma \ref{kernelestimate} to deduce that for $p>2$,
\begin{align}\label{Russoright}
  \|\mathcal{T}(F)\|_{S^{p,\infty}(L^2(\mathbb{R}^N,dw))}&\lesssim\left\|\frac{\|x-y\|}{d(x,y)}|K_{\D,j}(x,y)||F(x,y)|\right\|_{L_x^p(L_y^{p',\infty})^{\rm symm}}\nonumber\\&\lesssim \left\|\frac{|F(x,y)|}{w(B(x,d(x,y)))}\right\|_{L_x^p(L_y^{p',\infty})^{\rm symm}}.
\end{align}
By weak-type H\"{o}lder's inequality, for $1/q=1-2/p$,
\begin{align}\label{beforeinterpolation}
\left\|\frac{|F(x,y)|}{w(B(x,d(x,y)))}\right\|_{L_x^p(L_y^{p',\infty})}&\lesssim\left\|\frac{|F(x,y)|}{w(B(x,d(x,y)))^{2/p}}\right\|_{L_x^p(L_y^p)}\left\|\frac{1}{w(B(x,d(x,y)))^{1/q}}\right\|_{L_x^\infty( L_y^{q,\infty})}\nonumber\\
&\lesssim \|F\|_{L^p(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y))},
\end{align}
where the last inequality follows from \cite[Lemma 11.7]{HKarxiv}. Changing the role of $x$ and $y$ and then applying the doubling inequality \eqref{doublinginequality}, we deduce that
$$\left\|\frac{|F(x,y)|}{w(B(x,d(x,y)))}\right\|_{L_y^p(L_x^{p',\infty})}\lesssim \|F\|_{L^p(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y))}.$$
Therefore, the right-hand side of \eqref{Russoright} is also dominated by $\|F\|_{L^p(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y))}$.
This finishes the proof of the claim. To obtain the desired result, we apply the real interpolation argument. Given $p>2$, we choose $2<p_0<p<p_1<\infty$ and $\theta\in (0,1)$ such that $\frac{1}{p}=\frac{1-\theta}{p_0}+\frac{\theta}{p_1}$. We have
\begin{align}\label{beforetaking}
\|\mathcal{T}(F)\|_{S^{p}(L^2(\mathbb{R}^N,dw))}
&=\|\mathcal{T}(F)\|_{(S^{p_0,\infty}(L^2(\mathbb{R}^N,dw)),S^{p_1,\infty}(L^2(\mathbb{R}^N,dw)))_{\theta,p}}\nonumber\\
&\lesssim \|F\|_{(L^{p_0}(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y)),L^{p_1}(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y)))_{\theta,p}}\nonumber\\
&=\|F\|_{L^p(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y))}.
\end{align}
where the inequality above follows from the real interpolation theorem (see \cite[Theorem 3.11.8]{Bergh}) and the equalities follow from the real interpolation theorem for $S^p$ spaces (see \cite[Th\'{e}or\`{e}m 2]{MR296684}) and $L^p$ spaces (see \cite[Theorem 5.2.1]{Bergh}).

We consider
$$F_b(x,y):=\frac{d(x,y)}{\|x-y\|}(b(x)-b(y)).$$ We have  $$\|F_b\|_{L^p(\mathbb{R}^{2N},w(B(x,d(x,y)))^{-2}dw(x)dw(y))}=\|b\|_{B^p_{\D}(\mathbb{R}^N)},$$ and
$$\mathcal{T}(F_b)=[b,R_{\D,j}].$$
Taking $F=F_b$ in the inequality \eqref{beforetaking} yields the required inequality in Proposition \ref{upper bound}.
\end{proof}

\begin{lemma}\label{constructHPQ}
 Assume that $P$ and $Q$ are two $w$-measurable sets on $\mathbb{R}^N$ such that $P\subset Q$. Suppose that there exist constants $1<A<B<\infty$ such that
\begin{align}\label{twosidecond}
  Aw(P)\leq w(Q)\leq Bw(P).
\end{align}
There is a $w$-measurable  function $H_{P,Q}$ on $\mathbb{R}^N$ satisfying
\begin{enumerate}
  \item\label{111} $|H_{P,Q}(x)|\leq w(Q)^{-1/2}\chi_Q(x)$;\smallskip
  \item\label{222} $\int_Q H_{P,Q}(x)dw(x)=0$;\smallskip
  \item\label{333} $|(b)_P-(b)_Q|\approx w(Q)^{-1/2}|\langle b,H_{P,Q}\rangle_{L^2(\mathbb{R}^N,dw)}|$, where the implicit constants depending only on $A$ and $B$.
\end{enumerate}
\end{lemma}
\begin{proof}
Denote by $c_{A,B}:=\min\left\{(A-1)^{1/2},(B-1)^{-1/2}\right\}$ and let
\begin{align*}
H_{P,Q}:=c_{A,B}\left(\frac{w(Q\backslash P)^{1/2}}{w(P)^{1/2}w(Q)^{1/2}}\chi_P(x)-\frac{w(P)^{1/2}}{w(Q)^{1/2}w(Q\backslash P)^{1/2}}\chi_{Q\backslash P}(x)\right).
\end{align*}
Now we verify the function $H_{P,Q}$ satisfies all the required properties.

$\bullet $ {\bf Verification of \eqref{111}.}

It is obvious that $\supp H_{P,Q}\subset Q$. Next, we apply inequality \eqref{twosidecond} to see that
\begin{align}\label{twosdde}
(A-1)w(P)\leq w(Q\backslash P)=w(Q)-w(P)\leq (B-1)w(P).
\end{align}
Thus, $c_{A,B}w(Q\backslash P)^{1/2}\leq w(P)^{1/2}$ and $c_{A,B}w(P)^{1/2}\leq w(Q\backslash P)^{1/2}.$  This verifies property \eqref{111}.

$\bullet $ {\bf Verification of \eqref{222}.}

Note that
\begin{align*}
\int_Q H_{P,Q}(x)dw(x)&=c_{A,B}\int_Q \left(\frac{w(Q\backslash P)^{1/2}}{w(P)^{1/2}w(Q)^{1/2}}\chi_P(x)-\frac{w(P)^{1/2}}{w(Q)^{1/2}w(Q\backslash P)^{1/2}}\chi_{Q\backslash P}(x)\right)dw(x)\\
&=c_{A,B}\left(\frac{w(Q\backslash P)^{1/2}w(P)^{1/2}}{w(Q)^{1/2}}-\frac{w(P)^{1/2}w(Q\backslash P)^{1/2}}{w(Q)^{1/2}}\right)\\
&=0.
\end{align*}
This verifies property \eqref{222}.

$\bullet $ {\bf Verification of \eqref{333}.}

We first show a stronger equality, that is,
\begin{align}\label{strongequality}
c_{A,B}\big((b)_P-(b)_Q\big)=\langle b,H_{P,Q}\rangle_{L^2(\mathbb{R}^N,dw)} \frac{w(Q\backslash P)^{1/2}}{w(P)^{1/2}w(Q)^{1/2}}.
\end{align}
Indeed, we have
\begin{align*}
&{\rm RHS}\ {\rm of}\ \eqref{strongequality}\\
&=c_{A,B}\left(\frac{w(Q)-w(P)}{w(P)w(Q)}\int_P b(x)dw(x)-\frac{1}{w(Q)}\int_{Q\backslash P}b(x)dw(x)\right)\\
&=c_{A,B}\left(\fint_Pb(x)dw(x)-\frac{1}{w(Q)}\int_Pb(x)dw(x)-\bigg(\fint_Qb(x)dw(x)-\frac{1}{w(Q)}\int_Pb(x)dw(x)\bigg)\right)\\
&={\rm LHS}\ {\rm of}\ \eqref{strongequality}.
\end{align*}
Combining \eqref{strongequality} with \eqref{twosdde}, we deduce property \eqref{333} and then, Lemma \ref{constructHPQ}.
\end{proof}

\begin{lemma}\label{NWOlemma}
Let $b$ and $H$ be two $w$-measurable complex-valued functions on $\mathbb{R}^N$. Assume that there is a cube ${\displaystyle Q\in \bigcup_{\nu=1}^{2^N}\mathbb{D}^\nu}$ such that
\begin{enumerate}
  \item ${\displaystyle|H(x)|\leq w(Q)^{-1/2}\chi_Q\ { for}\ { all}\  x\in\mathbb{R}^N;}$
  \item
${\displaystyle\int_{Q}H(x)dw(x)=0}$.
\end{enumerate}
There exist function sequences $\{e_{Q,s}\}_{s=1}^4$ and $\{f_{Q,s}\}_{s=1}^4$ such that
$$|e_{Q,s}|+|f_{Q,s}|\leq w(Q)^{-1/2}\chi_{6\sqrt{N}Q},$$ and that
\begin{align}\label{yieldsss}
  |\langle b,H\rangle_{L^2(\mathbb{R}^N,dw)}|&\lesssim w(Q)^{1/2}\sum_{s=1}^4|\langle [b,R_{\D,j}]e_{Q,s},f_{Q,s}\rangle_{L^2(\mathbb{R}^N,dw)}|.
\end{align}
\end{lemma}
\begin{proof}
By Lemma \ref{nondegenerate}, for any $k\in\mathbb{Z}$ and $Q\in\mathbb{D}_k^\nu$, since $Q\subset B(c_Q,\sqrt{N}2^{-k-1})=:B_Q$,
there exists a ball $\widehat{B_Q}:=B(c_{\widehat{B_Q}},\sqrt{N}2^{-k-1})$ such that $\|c_Q-c_{\widehat{B_Q}}\|=5\sqrt{N}2^{-k-1}$, and that for every $(x,y)\in Q\times \widehat{B_Q}$, $K_{\D,j}(x,y)$ does not change sign and satisfies
 \begin{align}\label{nondegeneratelower}
  |K_{\D,j}(x,y)|\gtrsim\frac{1}{w(B_Q)}.
\end{align}

It follows from the doubling inequality \eqref{doublinginequality} that
$$w(B_Q)\sim w(\widehat{B_Q})\sim w(Q).$$

By \cite[Theorem 1.5]{Zhangwei2}, there exist $\theta\in [0,2\pi)$, depending on $\widehat{B_Q}$ and $b$, and $\alpha_{\widehat{B_Q}}(b)\in\mathbb{C}$ such that if we denote
$$F_s^Q:=\left\{x\in\widehat{B_Q}:\ -\frac{\pi}{4}+\frac{(s-1)\pi}{2}\leq \arg\Big(e^{i\theta}\big(\alpha_{\widehat{B_Q}}(b)-b(x)\big)\Big)\leq -\frac{\pi}{4}+\frac{s\pi}{2}\ or\ b(x)=\alpha_{\widehat{B_Q}}(b)\right\},$$
where $s\in\{1,2,3,4\}$, and $\arg(z)$ is the argument of a complex number $z$, then
$w(F_s^Q)\geq \frac{1}{16}w(\widehat{B_Q})$, and $F_1^Q\cup F_2^Q\cup F_3^Q\cup F_4^Q=\widehat{B_Q}$. $\alpha_{\widehat{B_Q}}(b)$ is called a complex median value of $b$ over $\widehat{B_Q}$.
Similarly, if we denote
\begin{align}
  E_s^Q:=\left\{x\in Q:\ -\frac{\pi}{4}+\frac{(s-1)\pi}{2}\leq \arg\Big(e^{i\theta}\big(b(x)-\alpha_{\widehat{B_Q}}(b)\big)\Big)\leq -\frac{\pi}{4}+\frac{s\pi}{2}\ or\ b(x)=\alpha_{\widehat{B_Q}}(b)\right\}
\end{align}
then $E_1^Q\cup E_2^Q\cup E_3^Q \cup E_4^Q= Q$.

It follows from the assumption of $H$ that
\begin{align*}
  |\langle b,H\rangle_{L^2(\mathbb{R}^N,dw)}|
  &=\Bigg|\int_Q b(x)H(x)dw(x)\Bigg|\nonumber\\
  &=\Bigg|\int_Q\big(b(x)-\alpha_{\widehat{B_Q}}(b)\big)H(x)dw(x)\Bigg|\nonumber\\
  &\lesssim w(Q)^{-\frac{1}{2}}\int_Q|b(x)-\alpha_{\widehat{B_Q}}(b)|dw(x)\nonumber\\
  &\lesssim w(Q)^{-\frac{1}{2}}\sum_{s=1}^4\int_{E_s^Q}|b(x)-\alpha_{\widehat{B_Q}}(b)|dw(x)=:\sum_{s=1}^4 M_s^Q.
\end{align*}
Note that for any $s\in\{1,2,3,4\},$ if $x\in E_s^Q$ and $y\in F_s^Q$, then
\begin{align}\label{complexmedian}
  |b(x)-\alpha_{\widehat{B_Q}}(b)|&\leq |e^{i\theta}(b(x)-\alpha_{\widehat{B_Q}}(b))|+|e^{i\theta}(\alpha_{\widehat{B_Q}}(b)-b(y))|\nonumber\\
  &\leq 2|e^{i\theta}(b(x)-b(y))|=2|b(x)-b(y)|.
\end{align}
Combining \eqref{nondegeneratelower} with \eqref{complexmedian}, we deduce that
\begin{align}
  M_s^Q&\lesssim w(Q)^{-\frac{1}{2}}\int_{E_s^Q}|b(x)-\alpha_{\widehat{B_Q}}(b)|dw(x)\cdot \frac{w(F_s^Q)}{w(Q)}\nonumber\\
  &\lesssim w(Q)^{-\frac{1}{2}}\int_{E_s^Q}\int_{F_s^Q}|b(x)-\alpha_{\widehat{B_Q}}(b)||K_{\D,j}(x,y)|dw(y)dw(x)\nonumber\\
  &\lesssim w(Q)^{-\frac{1}{2}}\int_{E_s^Q}\int_{F_s^Q}|b(x)-b(y)||K_{\D,j}(x,y)|dw(y)dw(x).
\end{align}
Moreover, note that for any $x\in E_1^Q$ and $y\in F_1^Q$, the arguments of $e^{i\theta}(b(x)-\alpha_{\widehat{B_Q}}(b))$ and $e^{i\theta}(\alpha_{\widehat{B_Q}}(b)-b(y))$ both belong to $[-\frac{\pi}{4},\frac{\pi}{4}]$. Thus,
$$-\frac{\pi}{4}\leq \arg\Big(e^{i\theta}\big(b(x)-b(y)\big)\Big)\leq \frac{\pi}{4}.$$
This implies that
\begin{align}
  |b(x)-b(y)|=\Big|e^{i\theta}\big(b(x)-b(y)\big)\Big|\leq 2{\rm Re}\Big(e^{i\theta}\big(b(x)-b(y)\big)\Big).
\end{align}
%and
%\begin{align}
%\Big|{\rm Im}\Big(e^{i\theta}(b(x)-b(y))\Big)\Big|\leq {\rm Re}\Big(e^{i\theta}\big(b(x)-b(y)\big)\Big).
%\end{align}
Therefore,
\begin{align*}
  M_1^Q
  &\lesssim w(Q)^{-\frac{1}{2}}\int_{E_1^Q}\int_{F_1^Q}{\rm Re}\Big(e^{i\theta}\big(b(x)-b(y)\big)\Big)K_{\D,j}(x,y)dw(y)dw(x)\nonumber\\
   &\lesssim w(Q)^{-\frac{1}{2}}\Bigg|\int_{E_1^Q}\int_{F_1^Q}e^{i\theta}\big(b(x)-b(y)\big)K_{\D,j}(x,y)dw(y)dw(x)\Bigg|\nonumber\\
   &= w(Q)^{-\frac{1}{2}}\Bigg|\int_{E_1^Q}\int_{F_1^Q}\big(b(x)-b(y)\big)K_{\D,j}(x,y)dw(y)dw(x)\Bigg|.
\end{align*}
The other three terms  $M_2^Q$, $M_3^Q$, $M_4^Q$ can be handled in the same way as $M_1^Q$ by rotation. Hence, for $s\in\{1,2,3,4\}$,
\begin{align}
  M_s^Q&\lesssim w(Q)^{-\frac{1}{2}}\Bigg|\int_{E_s^Q}\int_{F_s^Q}\big(b(x)-b(y)\big)K_{\D,j}(x,y)dw(y)dw(x)\Bigg|\nonumber\\
  &=w(Q)^{\frac{1}{2}}\Bigg\langle [b,R_{\D,j}]\frac{\chi_{F_s^Q}}{w(Q)^{1/2}},\frac{\chi_{E_s^Q}}{w(Q)^{1/2}}\Bigg\rangle_{L^2(\mathbb{R}^N,dw)}.
\end{align}
Letting
$$e_{Q,s}:=\frac{\chi_{F_s^Q}}{w(Q)^{1/2}}\  {\rm and}\  f_{Q,s}:=\frac{\chi_{E_s^Q}}{w(Q)^{1/2}},$$
which deduces \eqref{yieldsss}.
Clearly, $\supp E_s^Q\subset Q$. Moreover, since $\|c_Q-c_{\widehat{B_Q}}\|=5\sqrt{N}2^{-k-1}$, we see that $\supp F_s^Q\subset  6\sqrt{N}Q$. This completes the proof of Lemma \ref{NWOlemma}.
\end{proof}

\begin{lemma}\label{besovlemma}
For any $1\leq p<\infty$ and $b\in L^1_{\rm loc}(\mathbb{R}^N,dw)$, we have
  \begin{align}
  \|b\|_{B^{p}_{\D}(\mathbb{R}^N)}\lesssim \|b\|_{B^p(\mathbb{R}^N,dw)}+\|b\|_{B^{p}_{{\rm Non}-{\rm standard}}(\mathbb{R}^N)},
  \end{align}
  where we write
  $$\|b\|_{B^p(\mathbb{R}^N,dw)}:=\Bigg({\int_{\mathbb{R}^N}\int_{\mathbb{R}^N}\frac{|b(x)-b(y)|^p}{w(B(x,\|x-y\|))^2}}dw(x)dw(y)\Bigg)^{1/p}$$
  and
  \begin{align*}
  \|b\|_{B^{p}_{{\rm Non}-{\rm standard}}(\mathbb{R}^N)}
  &:=\Bigg(\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\sum_{k\in\mathbb{Z}}\sum_{{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|}}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\times\nonumber\\
  &\hspace{3.0cm}\times\fint_{B(c_Q,A_22^{-k+1})}\fint_{B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})}|b(x)-b(y)|^pdw(x)dw(y)\Bigg)^{1/p}.
  \end{align*}
\end{lemma}
\begin{proof}
Let $\Gamma$ be a  fixed Weyl chamber. By decomposing
$$\mathbb{R}^N=\bigcup_{\sigma\in G}\sigma(\Gamma),$$
we write
\begin{align}\label{besovdecom}
\|b\|_{B^{p}_{\D}(\mathbb{R}^N)}^p&=\sum_{\sigma,\widetilde{\sigma}\in G}\int_{\sigma(\Gamma)}\int_{\widetilde{\sigma}(\Gamma)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y)\nonumber\\
&=\sum_{\sigma\in G}\int_{\sigma(\Gamma)}\int_{\sigma(\Gamma)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y)\nonumber\\
&\hspace{1.0cm}+\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\int_{\sigma(\Gamma)}\int_{\widetilde{\sigma}(\Gamma)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y)\nonumber\\
&=\sum_{\sigma\in G}\int_{\sigma(\Gamma)}\int_{\sigma(\Gamma)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y)\nonumber\\
&\hspace{1.0cm}+\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\int_{\sigma(\Gamma)}\int_{D(y)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y)\nonumber\\
&\hspace{1.0cm}+\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\int_{\sigma(\Gamma)}\int_{E(y)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y),
\end{align}
where we denote by
$$D(y):=\left\{x\in \widetilde{\sigma}(\Gamma):\ \|x-y\|< 2^{10}d(x,y)\right\},$$
$$E(y):=\left\{x\in \widetilde{\sigma}(\Gamma):\ \|x-y\|\geq 2^{10}d(x,y)\right\}.$$

It follows from \eqref{distanceequa} that $d(x,y)=\|x-\widetilde{\sigma}(\sigma(y))\|$ if $x\in \widetilde{\sigma}(\Gamma)$ and $y\in \sigma(\Gamma)$ and that $d(x,y)\leq \|x-y\|$ for all $x,y\in\mathbb{R}^N$. In particular, to estimate the first term on the right-hand side of \eqref{besovdecom}, we note that for each $\sigma\in G$ and for every $x,y\in\sigma(\Gamma)$, $d(x,y)=\|x-y\|$. To estimate the second term, we apply the doubling inequality \eqref{doublinginequality} to
replace $d(x,y)$ by $\|x-y\|$. Consequently, the first two terms are dominated by $C\|b\|_{B^{p}(\mathbb{R}^N,dw)}$, where we used the facts that $G$ is a finite group and $w$ is a doubling measure. Thus, it remains to estimate the third term.

% we write all the elements of $G$ in a sequence $\sigma_1,\sigma_2,\ldots,\sigma_{|G|}$, and set $\Gamma_j:=\sigma_j(\Gamma)$.
Let ${\mathbb{D}_{\Gamma}:=\bigcup_{k\in\mathbb{Z}}\mathbb{D}_{\Gamma,k}}$ be a system of dyadic cubes over $\Gamma$, which is constructed in Corollary \ref{Construction over gamma}. It follows that ${ \sigma(\mathbb{D}_{\Gamma}):=\bigcup_{k\in\mathbb{Z}}\sigma(\mathbb{D}_{\Gamma,k})}$ is a system of dyadic cubes over $\sigma(\Gamma)$. Recall that $A_2$ is the constant given in Definition \ref{Defsystem}. We have
\begin{align}\label{rhs111}
&\int_{\sigma(\Gamma)}\int_{E(y)}\left(\frac{d(x,y)}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,d(x,y)))^2}dw(x)dw(y)\nonumber\\
&=\int_{\sigma(\Gamma)}\int_{E(y)}\left(\frac{\|x-\widetilde{\sigma}(\sigma(y))\|}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,\|x-\widetilde{\sigma}(\sigma(y))\|))^2}dw(x)dw(y)\nonumber\\
&\leq \sum_{k\in\mathbb{Z}}\int_{\sigma(\Gamma)}\int_{E_k(y)}\left(\frac{A_22^{-k}}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,A_22^{-k-1}))^2}dw(x)dw(y)\nonumber\\
&=\sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\int_{Q}\int_{E_k(y)}\left(\frac{A_22^{-k}}{\|x-y\|}\right)^p\frac{|b(x)-b(y)|^p}{w(B(x,A_22^{-k-1}))^2}dw(x)dw(y),
\end{align}
where
$$E_k(y):=\left\{x\in E(y):\ A_2 2^{-k-1}\leq \|x-\widetilde{\sigma}(\sigma(y))\|\leq A_22^{-k}\right\}.$$
It follows from the doubling inequality \eqref{doublinginequality} and the $G$-invariance of the Dunkl measure that for every $y\in Q$ and $x\in E_k(y)\subset B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})$ with $Q\in \sigma(\mathbb{D}_{\Gamma,k+10})$,
$$w(B(x,A_22^{-k-1}))\sim w(B(\widetilde{\sigma}(\sigma(c_Q)),2^{-k}))\sim w(B(c_Q,2^{-k}))\sim w(Q).$$
Substituting this inequality into \eqref{rhs111}, we deduce that
\begin{align}\label{rhs333}
  &{\rm RHS}\ {\rm of}\ \eqref{rhs111}\nonumber\\
  &\lesssim\sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\frac{2^{-kp}}{w(B(c_Q,2^{-k}))}\fint_{Q}\int_{E_k(y)}\left(\frac{|b(x)-b(y)|}{\|x-y\|}\right)^pdw(x)dw(y)\nonumber\\
    &\lesssim \sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\sum_{\ell\in\mathbb{Z}}\frac{2^{-kp}}{w(B(c_Q,2^{-k}))}\fint_{Q}\int_{E_{k,\ell}(y)}\left(\frac{|b(x)-b(y)|}{\|x-y\|}\right)^pdw(x)dw(y)\nonumber\\
    &\lesssim \sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\sum_{\ell\in\mathbb{Z}}\frac{2^{(\ell-k)p}}{w(B(c_Q,2^{-k}))}\fint_{Q}\int_{E_{k,\ell}(y)}|b(x)-b(y)|^pdw(x)dw(y),
\end{align}
where $$E_{k,\ell}(y):=\left\{x\in E_k(y):\ A_22^{-\ell-1}\leq \|x-y\|\leq A_22^{-\ell}\right\}.$$
Note that for every $y\in \sigma(\Gamma)$ and $x\in E_{k,\ell}(y)$,
$$A_22^{-k-1}\leq \|x-\widetilde{\sigma}(\sigma(y))\|=d(x,y)\leq 2^{-10}\|x-y\|\leq A_22^{-\ell-10}.$$
This implies that the summands on the right-hand side of \eqref{rhs333} vanish provided $\ell>k-9$. To continue, for every $Q\in \mathbb{D}_{\Gamma}$, we denote by $$U_k(Q):=B(c_Q,A_22^{-k+1})\backslash B(c_Q,A_22^{-k-2}).$$
Note that the reflection $\sigma$ is linear and preserves the Euclidean norm. Therefore, if $y\in Q\in\sigma(\mathbb{D}_{\Gamma,k+10})$ and $x\in E_{k,\ell}(y)$, then
$$\|x-\widetilde{\sigma}(\sigma(c_Q))\|\leq \|x-\widetilde{\sigma}(\sigma(y))\|+\|\widetilde{\sigma}(\sigma(y))-\widetilde{\sigma}(\sigma(c_Q))\|\leq A_22^{-k}+A_22^{-(k+10)}\leq A_22^{-(k+1)},$$
$$\|x-\widetilde{\sigma}(\sigma(c_Q))\|\geq \|x-\widetilde{\sigma}(\sigma(y))\|-\|\widetilde{\sigma}(\sigma(y))-\widetilde{\sigma}(\sigma(c_Q))\|\geq A_22^{-k-1}-A_22^{-(k+10)}\geq A_22^{-(k-2)},$$
$$\|x-c_Q\|\leq \|x-y\|+\|y-c_Q\|\leq A_22^{-\ell}+A_22^{-(k+10)}\leq A_22^{-\ell+1},$$
$$\|x-c_Q\|\geq \|x-y\|-\|y-c_Q\|\geq A_22^{-\ell-1}-A_22^{-(k+10)}\geq A_22^{-\ell-2}.$$
In other words, we have $x\in U_k(\widetilde{\sigma}(\sigma(Q)))\cap U_{\ell}(Q)$. Combining these facts, we see that
\begin{align}\label{rhs444}
&{\rm RHS}\ {\rm of}\ \eqref{rhs333}\nonumber\\
&=\sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\sum_{\ell\leq k-9}\frac{2^{(\ell-k)p}}{w(B(c_Q,2^{-k}))}\fint_{Q}\int_{E_{k,\ell}(y)}|b(x)-b(y)|^pdw(x)dw(y)\nonumber\\
&\lesssim \sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\sum_{\ell\leq k-9}\frac{2^{(\ell-k)p}}{w(B(c_Q,2^{-k}))}\fint_{Q}\int_{E_{k,\ell}}|b(x)-b(y)|^pdw(x)dw(y),
\end{align}
where
$$E_{k,\ell}:=\widetilde{\sigma}(\Gamma)\cap U_k(\widetilde{\sigma}(\sigma(Q)))\cap U_{\ell}(Q).$$

We {\it claim} that if $\ell\geq -\log_2(A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)+3$, then
\begin{align}\label{setclaim1}
B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})\cap B(c_Q,A_22^{-\ell+1})=\varnothing.
\end{align}
In particular, we have $U_k(\widetilde{\sigma}(\sigma(Q)))\cap U_{\ell}(Q)=\varnothing$. If this is not the case, then there exists an element $x_0\in B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})\cap B(c_Q,A_22^{-\ell+1})$. Equivalently, we have $\|x_0-\widetilde{\sigma}(\sigma(c_Q))\|\leq A_22^{-k+1}$ and $\|x_0-c_Q\|\leq A_22^{-\ell+1}$. Together with the inequality $\ell\leq k-9$, we see that
$$\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\leq \|x_0-\widetilde{\sigma}(\sigma(c_Q))\|+\|x_0-c_Q\|\leq A_22^{-k+1}+A_22^{-\ell+1}\leq A_22^{-\ell+2}.$$
It follows from the assumption $\ell\geq -\log_2(A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)+3$ that
$$A_22^{-\ell+2}\leq \frac{1}{2}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|.$$
Combining with the above two inequalities, we deduce that $c_Q=\widetilde{\sigma}(\sigma(c_Q))\in \sigma(\Gamma)\cap \widetilde{\sigma}(\Gamma)$, which  is a contradiction since each $c_Q$ does not lie in the boundaries of the Weyl chambers. This ends the proof of  \eqref{setclaim1}.

Next, we {\it claim} that if $\ell\leq \min\left\{k-4,-\log_2(A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)-3\right\}$, then
\begin{align}\label{setclaim2}
B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})\subset B(c_Q,A_22^{-\ell-2}).
\end{align}
In particular, we have $U_k(\widetilde{\sigma}(\sigma(Q)))\cap U_{\ell}(Q)=\varnothing$. Indeed, we take $x$ to be an arbitrary element of $B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})$, then
$$\|x-c_Q\|\leq \|x-\widetilde{\sigma}(\sigma(c_Q))\|+\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\leq A_22^{-k+1}+\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\leq A_22^{-\ell-2},$$
where the last inequality follows from the assumption. This ends the proof of \eqref{setclaim2}.

Denote by $$\ell_{k,Q}:=\min\left\{k-4,-\log_2(A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)-3\right\},$$
$$L_{k,Q}:=\min\left\{k-9,-\log_2(A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)+3\right\}.$$
Combining \eqref{setclaim1} with \eqref{setclaim2}, we see that
\begin{align}\label{rhs555}
&{\rm RHS}\ {\rm of}\ \eqref{rhs444}\nonumber\\
&=\sum_{k\in\mathbb{Z}}\sum_{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})}\sum_{\ell_{k,Q}<\ell\leq L_{k,Q}}\frac{2^{(\ell-k)p}}{w(B(c_Q,2^{-k}))}\fint_{Q}\int_{E_{k,\ell}}|b(x)-b(y)|^pdw(x)dw(y)\nonumber\\
&= \sum_{k\in\mathbb{Z}}\sum_{{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|}}}\sum_{-3<\ell+\log_2(A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)\leq 4}\frac{2^{(\ell-k)p}}{w(B(c_Q,2^{-k}))}\times\nonumber\\
&\hspace{9.0cm}\times\fint_{Q}\int_{E_{k,\ell}}|b(x)-b(y)|^pdw(x)dw(y).
\end{align}
Next, using the trivial inclusions
$E_{k,\ell}\subset B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})$ and $Q\subset B(c_Q,A_22^{-k+1})$,
the doubling inequality \eqref{doublinginequality} and the $G$-invariance of the Dunkl measure, we conclude that the right-hand side of \eqref{rhs555} is dominated by
$$C\sum_{k\in\mathbb{Z}}\sum_{{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|}}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\fint_{B(c_Q,A_22^{-k+1})}\fint_{B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})}|b(x)-b(y)|^pdw(x)dw(y).$$
This ends the proof of Lemma \ref{besovlemma}.
\end{proof}
With the collection $\{\mathbb{D}^\nu\}_{\nu=1}^{2^N}$ of dyadic cubes systems given in Lemma \ref{Nsystems}, we define the conditional expectation of a locally integrable function $f$ on $\mathbb{R}^N$ by the expression
$$E_k^\nu(f)(x):=\sum_{Q\in\mathbb{D}_k^\nu}(f)_Q\chi_Q(x),\ x\in\mathbb{R}^N.$$
Moreover, we define the martingale difference of $f$ by
\[
D_k^\nu (f):=E_{k+1}^\nu (f)-E_k^\nu (f).
\]

\begin{definition}
For any $1\leq p<\infty$ and $f\in L^1_{\rm loc}(\mathbb{R}^N,dw)$, we define the martingale Besov  (semi)norm of $f$ by
    $$\|f\|_{{MB}^p(\mathbb{R}^N,dw)}:=\sum_{\nu=1}^{2^N}\bigg(\sum_{k\in\mathbb{Z}}\int_{\mathbb{R}^{N}}\frac{|D_k^\nu f(x)|^p}{w(B(x,2^{-k}))}dw(x)\bigg)^{1/p}.$$
\end{definition}

\begin{lemma}\label{MartingaleBesov}
Suppose that $1\le p<\infty$ and $b\in L^1_{\mathrm{loc}}(\mathbb{R}^N,dw)$. We have
\[
\|b\|_{B^p(\mathbb{R}^N,dw)}\lesssim \|b\|_{{MB}^p(\mathbb{R}^N,dw)}.
\]
\end{lemma}
\begin{proof}
We first decompose $\mathbb{R}^N\times \mathbb{R}^N$ into dyadic shells as follows.
\begin{align}\label{rhs1010}
  \int_{\mathbb{R}^N}\int_{\mathbb{R}^N}\frac{|b(x)-b(y)|^p}{w(B(x,\|x-y\|))^2}dw(x)dw(y)
  &\leq \sum_{k\in\mathbb{Z}}\iint_{\substack{2^{-k-1}<\|x-y\|\leq 2^{-k}\\x,y\in\mathbb{R}^N}}\frac{|b(x)-b(y)|^p}{w(B(x,\|x-y\|))^2}dw(x)dw(y)\nonumber\\
  &\leq \sum_{k\in\mathbb{Z}}\iint_{\substack{2^{-k-1}<\|x-y\|\leq 2^{-k}\\x,y\in\mathbb{R}^N}}\frac{|b(x)-b(y)|^p}{w(B(x,2^{-k-1}))^2}dw(x)dw(y).
\end{align}
Fix $k\in\mathbb{Z}$ and let $\mathbb{D}_k^0$ be the standard system of dyadic cubes over $\mathbb{R}^N$ at scale $2^{-k}$ (see Notation \ref{standardsystem}).
For each $Q\in\mathbb{D}_k^0$, one has
$$I_Q:=\left\{y\in\mathbb{R}^N: {\rm dist}(y,Q)\leq 2^{-k}\right\}\subset B(c_Q,\sqrt{N}2^{-k+1}),$$
where we used the notation ${\rm dist}(y,Q)$ to denote the Euclidean distance from a point $y\in\mathbb{R}^N$ to a set $Q$. By Lemma \ref{Nsystems}, there exists ${\displaystyle Q'\in \bigcup_{\nu=1}^{2^N}\mathbb{D}_{k-k_0}^\nu}$, where $k_0:=\lfloor \log_2(24\sqrt{N})\rfloor$, such that
$$B(c_Q,\sqrt{N}2^{-k+1})\subset Q'\subset B(c_Q,12N\cdot2^{-k+1}).$$
With these observations, we conclude that
\begin{align}\label{RHS1111}
  &{\rm RHS}\ {\rm of}\ \eqref{rhs1010}\nonumber\\
  &\leq \sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^0}\int_{Q}\int_{I_Q}\frac{|b(x)-b(y)|^p}{w(B(x,2^{-k-1}))^2}dw(x)dw(y)\nonumber\\
   &\leq \sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^0}\int_{Q'}\int_{Q'}\frac{|b(x)-b(y)|^p}{w(B(x,2^{-k-1}))^2}dw(x)dw(y)\nonumber\\
   &\leq \sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_{k-k_0}^\nu}\int_{Q}\int_{Q}\frac{|b(x)-b(y)|^p}{w(B(x,2^{-k-1}))^2}dw(x)dw(y),
\end{align}
where in the last inequality we used the fact that each $Q'\in \mathbb{D}_{k-k_0}^\nu$ contains at most $c$ (an absolute constant) $Q\in\mathbb{D}_k^0$. By the doubling inequality \eqref{doublinginequality}, for any $x,y\in Q\in\mathbb{D}_{k-k_0}^\nu$,
$$w(Q)\sim w(B(x,2^{-k-1}))\sim w(B(y,2^{-k-1})).$$
Therefore, by inserting and subtracting the conditional expectation $E_{k-k_0}^\nu(b)$, we deduce that
\begin{align}
  &{\rm RHS}\ {\rm of}\ \eqref{RHS1111}\nonumber\\
  &\lesssim \sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_{k-k_0}^\nu}\int_{Q}\int_{Q}\frac{|b(x)-E_{k-k_0}^\nu(b)(x)|^p}{w(B(x,2^{-k-1}))^2}+\frac{|b(y)-E_{k-k_0}^\nu(b)(y)|^p}{w(B(x,2^{-k-1}))^2}dw(x)dw(y)\nonumber\\
  &\lesssim\sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_{k-k_0}^\nu}\int_{Q}\frac{|b(x)-E_{k-k_0}^\nu(b)(x)|^p}{w(B(x,2^{-k-1}))}dw(x)\nonumber\\
  &\lesssim\sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\int_{\mathbb{R}^N}\frac{|b(x)-E_{k}^\nu(b)(x)|^p}{w(B(x,2^{-k}))}dw(x),
\end{align}
where the last inequality used the change of index and the doubling inequality \eqref{doublinginequality}.

To proceed, we apply  Lemma~\ref{Haarexpansion} to write
\[
b(x)-E_k^\nu (b)(x)=\sum_{m=0}^{\infty}D_{k+m}^\nu (b)(x),
\]
 where the sum converges pointwise almost everywhere. Moreover, by the lower doubling inequality \eqref{doublinginequality},
\begin{align*}
C^{-1}2^{mN}\leq \frac{w(B(x,2^{-k}))}{w(B(x,2^{-(m+k)}))}.
\end{align*}
Combining these facts, we deduce that
\begin{align}\label{eq:A-k-new}
\Bigg(\int_{\mathbb{R}^N}\frac{|b(x)-E_k^\nu (b)(x)|^p}{w(B(x,2^{-k}))}\,dw(x)\Bigg)^{\!1/p}
&\le \sum_{m\ge0}
\Bigg(\int_{\mathbb{R}^N}\frac{|D_{k+m}^\nu (b)(x)|^p}{w(B(x,2^{-k}))}\,dw(x)\Bigg)^{\!1/p}\nonumber\\
&\lesssim \sum_{m\ge0}2^{-mN/p}\Bigg(\int_{\mathbb{R}^N}\frac{|D_{k+m}^\nu (b)(x)|^p}{w(B(x,2^{-(k+m)}))}\,dw(x)\Bigg)^{\!1/p}.
\end{align}
Since $\sum_{m\geq 0}2^{-mN/p}=(1-2^{-N/p})^{-1}<+\infty$, by discrete convolution Young's inequality, we have
\[
\Bigg(\sum_{k\in\mathbb{Z}}\int_{\mathbb{R}^N}\frac{|b(x)-E_k^\nu (b)(x)|^p}{w(B(x,2^{-k}))}\,dw(x)\Bigg)^{\!1/p}
\lesssim
\Bigg(\sum_{k\in\mathbb{Z}}\int_{\mathbb{R}^N}\frac{|D_{k}^\nu(b)(x)|^p}{w(B(x,2^{-k}))}\,dw(x)\Bigg)^{\!1/p}.
\]
Summing over $\nu$ yields the required assertion.
\end{proof}






\begin{lemma}\label{standard}
Suppose that $1\leq p<\infty$ and $b\in L^1_{\rm loc}(\mathbb{R}^N,dw)$. If $[b,R_{\D,j}]\in S^p$ for some $j\in\{1,2,\ldots,N\}$, then
\[
\|b\|_{B^p(\mathbb{R}^N,dw)}\lesssim \|[b,R_{\D,j}]\|_{S^p}.
\]
\end{lemma}
\begin{proof}
By Lemma \ref{MartingaleBesov}, it suffices to show that
for each $\nu\in\{1,2,\ldots,2^N\}$, we have
\begin{align}\label{martingaleassertion}
  \bigg(\sum_{k\in\mathbb{Z}}\int_{\mathbb{R}^N}\frac{|D_k^\nu(b)(x)|^p}{w(B(x,2^{-k}))}dw(x)\bigg)^{1/p}\lesssim\|[b,R_{\D,j}]\|_{S^p}.
\end{align}

  For any $Q\in\mathbb{D}_k^\nu$, we let $h_Q^1,h_Q^2,\ldots,h_Q^{2^N-1}$ be a family of Haar functions as defined in Section \ref{HaarSection}. Next, we choose $h_Q$ among these functions such that
  $$h_Q:=\left\{h_Q^i:\ \left|\int_Qb(x)h_Q^i(x)dw(x)\right|\ {\rm is}\ {\rm maximal}\ {\rm with}\ {\rm respect}\ {\rm to}\ i=1,2,\dots,2^{N}-1\right\}.$$
Note that for every $x\in Q$, $$D_k^{\nu}(b)(x)=\sum_{i=1}^{2^N-1}\langle b,h_Q^i\rangle h_Q^i(x),\ {\rm with}\ \|h_Q^i\|_{\infty}\lesssim w(Q)^{-1/2}.$$ Hence, we have
\begin{align}
  \bigg(\fint_Q |D_k^\nu(b)(x)|^pdw(x)\bigg)^{1/p}\leq C_{p,N} w(Q)^{-1/2}\bigg|\int_Qb(x)h_Q(x)dw(x)\bigg|
\end{align}
for some constant $C_{p,N}>0$.
This, together with the doubling inequality \eqref{doublinginequality}, implies
\begin{align}\label{hhhhhh}
  \int_{\mathbb{R}^N}\frac{|D_k^\nu(b)(x)|^p}{w(B(x,2^{-k}))}dw(x)
  &\lesssim \sum_{Q\in\mathbb{D}_k^\nu}\fint_Q|D_k^\nu(b)(x)|^pdw(x)\nonumber\\
  &\lesssim \sum_{Q\in\mathbb{D}_k^\nu}w(Q)^{-p/2}|\langle b,h_Q\rangle_{L^2(\mathbb{R}^N,dw)}|^p.
\end{align}
By Lemma \ref{NWOlemma}, there exist NWO sequences $\{u_{Q}^{(s)}\}$ and $\{v_{Q}^{(s)}\}$, $s=1,2,3,4$, such that
\begin{align}\label{yieldssss}
  |\langle b,h_Q\rangle_{L^2(\mathbb{R}^N,dw)}|&\lesssim w(Q)^{1/2}\sum_{s=1}^4|\langle [b,R_{\D,j}]u_{Q}^{(s)},v_{Q}^{(s)}\rangle_{L^2(\mathbb{R}^N,dw)}|.
\end{align}
Substituting \eqref{yieldssss} into \eqref{hhhhhh} and then applying Lemma \ref{NWOLEMMAA}, we deduce that
\begin{align*}
  \sum_{k\in\mathbb{Z}}\int_{\mathbb{R}^N}\frac{|D_k^\nu(b)(x)|^p}{w(B(x,2^{-k}))}dw(x)
  &\lesssim \sum_{s=1}^4\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}|\langle [b,R_{\D,j}]u_{Q}^{(s)},v_{Q}^{(s)}\rangle_{L^2(\mathbb{R}^N,dw)}|^p\nonumber\\
  &\lesssim \|[b,R_{\D,j}]\|_{S^p}^p.
\end{align*}
This shows the inequality \eqref{martingaleassertion} and then finishes the proof of Lemma \ref{standard}.
\end{proof}


\begin{lemma}\label{traceequality}
Let $T$ be a bounded linear operator on a Hilbert space $H$.
For any $\xi,\eta\in H$, we have
\[
\langle T\xi, \eta\rangle \;=\; \operatorname{Tr}\big((\xi\otimes \eta)T\big),
\]
where $\xi\otimes \eta$ denotes the rank-one operator defined by
\((\xi\otimes \eta)(x) := \langle x,\eta\rangle \xi\) for $x\in H$.
\end{lemma}

\begin{proof}
For any $x\in H$, we have
\[
(\xi\otimes \eta)T(x) = (\xi\otimes \eta)(Tx) = \langle Tx, \eta\rangle \xi=\langle x, T^*\eta\rangle \xi=(\xi\otimes T^*\eta)(x).
\]
Hence,
\[
(\xi\otimes \eta)T = \xi\otimes T^*\eta.
\]
Since the trace of a rank-one operator $u\otimes v$ is given by
\(\operatorname{Tr}(u\otimes v) = \langle u, v\rangle\), we obtain
\[
\operatorname{Tr}((\xi\otimes \eta)T) = \operatorname{Tr}(\xi\otimes T^*\eta) = \langle \xi, T^*\eta\rangle.
\]
Finally, $\langle \xi, T^*\eta\rangle = \langle T\xi, \eta\rangle$, which completes the proof.
\end{proof}


\begin{lemma}\label{nonstandard}
Suppose that $N< p<\infty$ and $b\in L^1_{\rm loc}(\mathbb{R}^N,dw)$. Assume that $[b,R_{\D,j}]\in S^p$ for some $j\in\{1,2,\ldots,N\}$, then we have
\begin{align*}
\|b\|_{B^{p}_{{\rm Non}-{\rm standard}}(\mathbb{R}^N)}\lesssim \|[b,R_{\D,j}]\|_{S^p}.
\end{align*}
\end{lemma}
\begin{proof}
Let $k_0:=\lfloor \log_2(24A_2) \rfloor$ be an integer such that $\frac{2^{-(k-k_0)-2}}{3}\leq A_22^{-k+1}<\frac{2^{-(k-k_0)-1}}{3}$.
By Lemma \ref{Nsystems}, for each $k\in \mathbb{Z}$ and $Q\in\sigma(\mathbb{D}_{\Gamma,k+10})$, there exist  cubes $W_Q,P_Q\in {\displaystyle\bigcup_{\nu=1}^{2^N}\mathbb{D}_{k-k_0}^\nu}$ such that
\begin{align}\label{Bcontain1}
B(c_Q,A_22^{-k+1})\subset W_Q\subset B(c_Q,12\sqrt{N} A_22^{-k+1}),
\end{align}
\begin{align}\label{Bcontain2}
  B(\widetilde{\sigma}(\sigma(c_Q)),A_22^{-k+1})\subset P_Q\subset B(\widetilde{\sigma}(\sigma(c_Q)),12\sqrt{N} A_22^{-k+1}).
\end{align}
Together with the doubling inequality \eqref{doublinginequality}, we deduce that
\begin{align*}
&\|b\|_{B^{p}_{{\rm Non}-{\rm standard}}(\mathbb{R}^N)}^p\nonumber\\
&\lesssim \sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\sum_{k\in\mathbb{Z}}\sum_{{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|}}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\fint_{W_Q}\fint_{P_Q}|b(x)-b(y)|^pdw(x)dw(y)\\
&=\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\sum_{\nu,\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ P_Q\in\mathbb{D}_{k-k_0}^\nu,W_Q\in\mathbb{D}_{k-k_0}^\tau }}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\fint_{W_Q}\fint_{P_Q}|b(x)-b(y)|^pdw(x)dw(y)\nonumber\\
&\lesssim \sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\sum_{\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ W_Q\in\mathbb{D}_{k-k_0}^\tau }}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\fint_{W_Q}|b(x)-(b)_{W_Q}|^pdw(x)\nonumber\\
&\hspace{1.0cm}+\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ P_Q\in\mathbb{D}_{k-k_0}^\nu}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\fint_{P_Q}|b(x)-(b)_{P_Q}|^pdw(x)\nonumber\\
&\hspace{1.0cm}+\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}\sum_{\nu,\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ P_Q\in\mathbb{D}_{k-k_0}^\nu,W_Q\in\mathbb{D}_{k-k_0}^\tau }}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{P_Q}-(b)_{W_Q}|^p\nonumber\\
&=:\sum_{{\substack{\sigma,\widetilde{\sigma}\in G \\ \sigma\neq \widetilde{\sigma}}}}({\rm I}_{\sigma,\tilde{\sigma}}^p+{\rm II}_{\sigma,\tilde{\sigma}}^p+{\rm III}_{\sigma,\tilde{\sigma}}^p).
\end{align*}
Since $Q\subset W_Q$ for all $Q\in\sigma(\mathbb{D}_{\Gamma,k+10})$, we see that for each $\widetilde{Q}\in \mathbb{D}_{k-k_0}^\tau$, there are at most finite (an absolute constant) $Q\in \sigma(\mathbb{D}_{\Gamma,k+10})$ such that $W_Q=\widetilde{Q}$. This implies that
\begin{align}
  {\rm I}_{\sigma,\tilde{\sigma}}^p
  &\lesssim \sum_{\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ W_Q\in\mathbb{D}_{k-k_0}^\tau }}\fint_{W_Q}|b(x)-(b)_{W_Q}|^pdw(x)\nonumber\\
    &\lesssim \sum_{\tau=1}^{2^N}\sum_{Q\in\mathbb{D}_{k-k_0}^\tau }\fint_{Q}|b(x)-(b)_{Q}|^pdw(x)\nonumber\\
    &\lesssim \|b\|_{B^p(\mathbb{R}^N,dw)}^p,
\end{align}
where in the last inequality we used the fact that oscillation space ${\rm OSC}^{p,p}(\mathbb{R}^N,dw)$ is equivalent to Besov space $B^p(\mathbb{R}^N,dw)$ (see e.g., \cite[Proposition 17.5]{HKarxiv}). Similarly, the second term ${\rm II}_{\sigma,\tilde{\sigma}}$ is also bounded by $\|b\|_{B^p(\mathbb{R}^N,dw)}$. Together with Lemma \ref{standard}, we deduce that
$${\rm I}_{\sigma,\tilde{\sigma}}+{\rm II}_{\sigma,\tilde{\sigma}}\lesssim \|[b,R_{\D,j}]\|_{S^p}.$$

Now we turn to the term ${\rm III}_{\sigma,\tilde{\sigma}}$. For each ${\displaystyle Q\in \bigcup_{\nu=1}^{2^N}\mathbb{D}_{k-k_0}^\nu}$ and $k\in\mathbb{N}$, denote by $Q^{[k]}$ the $k$th strict dyadic ancestor. For convenience, we also set $Q^{[0]}:=Q$.  Pick $\alpha(Q)\in\mathbb{Z}$ such that
$2^{\alpha(Q)}\leq A_2^{-1}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|<2^{\alpha(Q)+1}.$ Moreover, we let
$$s_Q:=\log_2(2^{k+10+\alpha(Q)})=k+10+\alpha(Q).$$
If $A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|$, then $s_Q$ is a positive integer larger than $10$. We have
\begin{align*}
{\rm III}_{\sigma,\tilde{\sigma}}^p
&\lesssim \sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ P_Q\in\mathbb{D}_{k-k_0}^\nu}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{P_Q}-(b)_{P_Q^{[s_Q]}}|^p\nonumber\\
&\hspace{1.0cm}+ \sum_{\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ W_Q\in\mathbb{D}_{k-k_0}^\tau}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{W_Q}-(b)_{W_Q^{[s_Q]}}|^p\nonumber\\
&\hspace{1.0cm}+\sum_{\nu,\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ P_Q\in\mathbb{D}_{k-k_0}^\nu,W_Q\in\mathbb{D}_{k-k_0}^\tau }}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{P_Q^{[s_Q]}}-(b)_{W_Q^{[s_Q]}}|^p\nonumber\\
&=:{\rm III}_{\sigma,\tilde{\sigma},1}^p+{\rm III}_{\sigma,\tilde{\sigma},2}^p+{\rm III}_{\sigma,\tilde{\sigma},3}^p.
\end{align*}
By Lemma \ref{constructHPQ}, for each $m\in\mathbb{N}$ and for each $Q\in\sigma(\mathbb{D}_{\Gamma})$ with $P_Q\in\mathbb{D}_{k-k_0}^\nu$, there is a $w$-measurable function $H_{Q,m}$ on $\mathbb{R}^N$ (alternatively, $H_{Q,m}$ can be taken to be the Haar Basis element $h_{P_Q^{[m]}}^1$ (up to an absolute constant) if we enumerate $Ch(P_Q^{[m]})=\{R_1,R_2,\ldots,R_{2^N}\}$ so that $R_1=P_Q^{[m-1]}$) satisfying
\begin{enumerate}
  \item $|H_{Q,m}(x)|\leq w(P_Q^{[m]})^{-1/2}\chi_{P^{[m]}_Q}(x)$;\smallskip
  \item $\int_{P_Q^{[m]}} H_{Q,m}(x)dw(x)=0$;\smallskip
  \item $|(b)_{P_Q^{[m-1]}}-(b)_{P_Q^{[m]}}|\approx w(P_Q^{[m]})^{-1/2}|\langle b,H_{Q,m}\rangle_{L^2(\mathbb{R}^N,dw)}|$.
\end{enumerate}
For simplicity, we denote by
$$\Lambda^\nu:=\left\{(k,Q,m):k\in\mathbb{Z}, Q\in\mathcal{\sigma}(\mathbb{D}_{\Gamma,k+10}),A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|,P_Q\in\mathbb{D}_{k-k_0}^\nu,1\leq m\leq s_Q\right\}.$$
By Minkowski's inequality,
\begin{align}\label{rhs666}
{\rm III_{\sigma,\tilde{\sigma},1}}
&\lesssim \sum_{\nu=1}^{2^N}\Bigg(\sum_{(k,Q,m)\in\Lambda^\nu}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{P_Q^{[m-1]}}-(b)_{P_Q^{[m]}}|^p\Bigg)^{1/p}\nonumber\\
&\lesssim \sum_{\nu=1}^{2^N}\Bigg(\sum_{(k,Q,m)\in\Lambda^\nu}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}w(P_Q^{[m]})^{-p/2}|\langle b,H_{Q,m}\rangle_{L^2(\mathbb{R}^N,dw)}|^p\Bigg)^{1/p}.
\end{align}
By Lemma \ref{NWOlemma}, there exist NWO sequences $\{e_{P_Q^{[m]}}^{(s)}\}$ and $\{f_{P_Q^{[m]}}^{(s)}\}$, $s=1,2,3,4$, such that
\begin{align*}
w(P_Q^{[m]})^{-1/2}|\langle b,H_{Q,m}\rangle_{L^2(\mathbb{R}^N,dw)}|\lesssim \sum_{s=1}^4\big|\big\langle [b,R_{\D,j}]e_{P_Q^{[m]}}^{(s)},f_{P_Q^{[m]}}^{(s)}\big\rangle_{L^2(\mathbb{R}^N,dw)}\big|.
\end{align*}
Thus, the right-hand side of \eqref{rhs666} is dominated by
$$\sum_{\nu=1}^{2^N}\sum_{s=1}^4\Bigg(\sum_{(k,Q,m)\in\Lambda^\nu}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}\big|\big\langle [b,R_{\D,j}]e_{P_Q^{[m]}}^{(s)},f_{P_Q^{[m]}}^{(s)}\big\rangle_{L^2(\mathbb{R}^N,dw)}\big|^p\Bigg)^{1/p}.$$
Now taking $\{\varphi_{k,Q,m}\}_{(k,Q,m)\in\Lambda^\nu}$ with unit $\ell^{p'}$ norm, we apply the duality argument to estimate
\begin{align}\label{repeatduality}
&\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\big\langle [b,R_{\D,j}]e_{P_Q^{[m]}}^{(s)},f_{P_Q^{[m]}}^{(s)}\big\rangle_{L^2(\mathbb{R}^N,dw)}\varphi_{k,Q,m}\nonumber\\
&=\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}{\rm Tr}\Big(\big(e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)}\big)[b,R_{\D,j}]\Big)\varphi_{k,Q,m}\nonumber\\
&={\rm Tr}\Bigg(\Big(\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\varphi_{k,Q,m} (e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)})\Big)\cdot [b,R_{\D,j}]\Bigg)\nonumber\\
&\leq \left\|\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\varphi_{k,Q,m} (e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)})\right\|_{S^{p'}}\cdot \|[b,R_{\D,j}]\|_{S^p},
\end{align}
where the first equality follows from Lemma \ref{traceequality}.
It remains to show that
\begin{align}\label{supinequality}
\sup_{\|\{\varphi_{k,Q,m}\}_{(k,Q,m)\in\Lambda^\nu}\|_{\ell^{p'}}=1}\left\|\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\varphi_{k,Q,m} (e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)})\right\|_{S^{p'}}\lesssim 1.
\end{align}
To show \eqref{supinequality}, we first write
\begin{align}\label{rhs777}
&\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\varphi_{k,Q,m} (e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)})\nonumber\\
&=\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{Q\in \Theta_{m,k}^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\varphi_{k,Q,m} (e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)}),
\end{align}
where
$$\Theta_{m,k}^\nu:=\left\{Q\in\sigma(\mathbb{D}_{\Gamma,k+10}):A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|,\ \alpha(Q)\geq -k+m-10,\ P_Q\in\mathbb{D}_{k-k_0}^\nu\right\}.$$
Note that for any $Q\in \mathbb{D}_{k-k_0-m}^\nu$, there are at most $c2^{mN}$ cubes $W\in\sigma(\mathbb{D}_{\Gamma,k+10})$ such that $P_W^{[m]}=Q$, where $c$ is an absolute constant. Denote these cubes by $Q_{[m]}^{1}$, $Q_{[m]}^{2}$, $\ldots$, $Q_{[m]}^{\iota}$ with $\iota\leq c2^{mN}$. Let
$\delta_{Q,m,\ell}:=1$ if $P_{Q_{[m]}^\ell}=Q$ and $\delta_{Q,m,\ell}:=0$ otherwise. Let
$$\Xi_{m,k,\ell}^\nu:=\left\{Q\in \mathbb{D}_{k-k_0-m}^\nu: A_22^{-k+1}\leq \|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|,\ \alpha({Q_{[m]}^\ell})\geq -k+m-10\right\}.$$
We have
\begin{align}\label{rhs888}
&{\rm RHS}\ {\rm of}\ \eqref{rhs777}\nonumber\\
&=\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{\ell=1}^{c2^{mN}}\sum_{Q\in \Xi_{m,k,\ell}^\nu}2^{-k}\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|^{-1}\varphi_{k,{Q_{[m]}^\ell},m}\delta_{Q,m,\ell} (e_Q^{(s)}\otimes f_Q^{(s)})\nonumber\\
&=2^{-k_0}\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{\ell=1}^{c2^{mN}}\sum_{Q\in \Xi_{m,k+k_0+m,\ell}^\nu}2^{-(k+m)}\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|^{-1}\varphi_{k+k_0+m,{Q_{[m]}^\ell},m}\delta_{Q,m,\ell} (e_Q^{(s)}\otimes f_Q^{(s)})\nonumber\\
&=\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}\lambda_Q(e_Q^{(s)}\otimes f_Q^{(s)}),
\end{align}
where for any $Q\in \mathbb{D}_k^\nu$,
\begin{align*}
\lambda_Q&:=2^{-k_0}\sum_{m\geq 1}\sum_{\ell=1}^{c2^{mN}}2^{-(k+m)}\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|^{-1}\varphi_{k+k_0+m,{Q_{[m]}^\ell},m}\delta_{Q,m,\ell}\times\nonumber\\
&\hspace{1.0cm}\times\chi_{[A_22^{-(k+k_0+m-1)},\infty)}(\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|) \chi_{[-(k+k_0-10),\infty)}(\alpha(Q_{[m]}^\ell)).
\end{align*}
It follows from equalities \eqref{rhs777}, \eqref{rhs888} and Lemma \ref{NWOLEMMAA2} that
\begin{align}\label{rhs999}
&\left\|\sum_{(k,Q,m)\in\Lambda^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\varphi_{k,Q,m} (e_{P_Q^{[m]}}^{(s)}\otimes f_{P_Q^{[m]}}^{(s)})\right\|_{S^{p'}}\nonumber\\
&=\left\|\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}\lambda_Q(e_Q^{(s)}\otimes f_Q^{(s)})\right\|_{S^{p'}}\nonumber\\
&\lesssim \Bigg(\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}|\lambda_Q|^{p'}\Bigg)^{\frac{1}{p'}}.
\end{align}
By H\"{o}lder's inequality and the definition of $\lambda_Q$,
\begin{align*}
&\Bigg(\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}|\lambda_Q|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&\lesssim \Bigg(\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu} \bigg(\sum_{m\geq 1}\sum_{\ell=1}^{c2^{mN}}2^{-(k+m)p}\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|^{-p}\delta_{Q,m,\ell}\chi_{[-(k+k_0-10),\infty)}(\alpha(Q_{[m]}^\ell))\bigg)^{\frac{p'}{p}}\times\nonumber\\
&\hspace{1.0cm}\times \bigg(\sum_{m\geq 1}\sum_{\ell=1}^{c2^{mN}}\Big|\varphi_{k+k_0+m,{Q_{[m]}^\ell},m}\delta_{Q,m,\ell}\chi_{[A_22^{-(k+k_0+m-1)},\infty)}(\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|) \chi_{[-(k+k_0-10),\infty)}(\alpha(Q_{[m]}^\ell))\Big|^{p'}\bigg)    \Bigg)^{\frac{1}{p'}}.
\end{align*}
Since $p>N$, we have
\begin{align*}
&\sum_{m\geq 1}\sum_{\ell=1}^{c2^{mN}}2^{-(k+m)p}\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|^{-p}\delta_{Q,m,\ell}\chi_{[-(k+k_0-10),\infty)}(\alpha(Q_{[m]}^\ell))\nonumber\\
&\lesssim\sum_{m\geq 1}\sum_{\ell=1}^{c2^{mN}}2^{-(k+m)p}\cdot 2^{-kp}=C\sum_{m\geq 1}2^{-m(p-N)}\lesssim 1.
\end{align*}
Combining this with the same reindexing and counting argument used to derive \eqref{rhs888}, we obtain
\begin{align}\label{longdisplay}
&\Bigg(\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}|\lambda_Q|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&\lesssim \Bigg(\sum_{k\in\mathbb{Z}}\sum_{Q\in\mathbb{D}_k^\nu}\sum_{m\geq 1}\sum_{\ell=1}^{c2^{mN}}\Big|\varphi_{k+k_0+m,{Q_{[m]}^\ell},m}\times\nonumber\\
&\hspace{1.0cm}\times\delta_{Q,m,\ell}\chi_{[A_22^{-(k+k_0+m-1)},\infty)}(\|c_{Q_{[m]}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{[m]}^\ell}))\|) \chi_{[-(k+k_0-10),\infty)}(\alpha(Q_{[m]}^\ell))\Big|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&=\Bigg(\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{\ell=1}^{c2^{mN}}\sum_{Q\in\Xi_{m,k+k_0+m,\ell}^\nu}\Big|\varphi_{k+k_0+m,{Q_{[m]}^\ell},m}\Big|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&=\Bigg(\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{\ell=1}^{c2^{mN}}\sum_{Q\in\Xi_{m,k,\ell}^\nu}\Big|\varphi_{k,{Q_{[m]}^\ell},m}\Big|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&=\Bigg(\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{Q\in \Theta_{m,k}^\nu}|\varphi_{k,Q,m}|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&=\Bigg(\sum_{(k,Q,m)\in\Lambda^\nu}|\varphi_{k,Q,m}|^{p'}\Bigg)^{\frac{1}{p'}}\nonumber\\
&= 1.
\end{align}
Substituting this inequality into \eqref{rhs999}, we obtain the desired inequality \eqref{supinequality}. Therefore, ${\rm III}_{\sigma,\tilde{\sigma},1}$ is dominated by $C\|[b,R_{\D,j}]\|_{S^p}$. Similarly,  ${\rm III}_{\sigma,\tilde{\sigma},2}$ is also dominated by $C\|[b,R_{\D,j}]\|_{S^p}$.

Next, we turn to the term ${\rm III}_3$. Set $B_Q:=B(c_Q,(2^{k_0+11}+12\sqrt{N}+1)\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|)$. We {\it claim} that
\begin{align}\label{claiminclusion}
  W_Q^{[s_Q]}\cup P_Q^{[s_Q]}\subset B_Q.
\end{align}
Indeed, we first note that for any $Q\in\sigma(\mathbb{D}_{\Gamma,k+10})$ with $ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|$ and with $ P_Q\in\mathbb{D}_{k-k_0}^\nu,W_Q\in\mathbb{D}_{k-k_0}^\tau$, we have
\begin{align}\label{similarcombine}
  \ell(P_Q^{[s_Q]})=\ell(W_Q^{[s_Q]})=2^{k_0+10+\alpha(Q)}\leq\frac{2^{k_0+10}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|}{A_2}.
\end{align}
Combining this inequality with the inclusions that $W_{Q}^{[s_Q]}\subseteq B(c_{W_Q^{[s_Q]}},A_2\ell(W_Q^{[s_Q]}))$ and \eqref{Bcontain1}, we deduce that for any $x\in W_{Q}^{[s_Q]}$,
\begin{align*}
\|x-c_Q\|
&\leq \|x-c_{W_Q^{[s_Q]}}\|+\|c_{W_Q^{[s_Q]}}-c_{W_Q}\|+\|c_{W_Q}-c_Q\|\\
&\leq 2A_2\ell(W_Q^{[s_Q]})+12\sqrt{N}A_22^{-k+1}\\
&\leq (2^{k_0+11}+12\sqrt{N}) \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|.
\end{align*}
Similarly, combining \eqref{similarcombine} with the inclusions $P_Q^{[s_Q]}\subseteq B(c_{P_Q^{[s_Q]}},A_2\ell(P_Q^{[s_Q]}))$ and \eqref{Bcontain2}, we deduce that for any $x\in P_Q^{[s_Q]}$,
\begin{align*}
  \|x-c_Q\|
  &\leq \|x-c_{P_Q^{[s_Q]}}\|+\|c_{P_Q^{[s_Q]}}-c_{P_Q}\|+\|c_{P_Q}-\widetilde{\sigma}(\sigma(c_Q))\|+\|\widetilde{\sigma}(\sigma(c_Q))-c_Q\|\\
  &\leq 2A_2\ell(P_Q^{[s_Q]})+12\sqrt{N}A_22^{-k+1}+\|\widetilde{\sigma}(\sigma(c_Q))-c_Q\|\\
  &\leq (2^{k_0+11}+12\sqrt{N}+1) \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|
\end{align*}
This finishes the proof of \eqref{claiminclusion}.

%&\leq \kappa_NA_22^{-k+1}+\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|+\kappa_NA_22^{-k+1}\\

By Lemma \ref{Nsystems}, for any $Q\in\sigma(\mathbb{D}_{\Gamma})$, there exists a cube $E_Q\in {\displaystyle\bigcup_{\nu=1}^{2^N}\mathbb{D}^\nu}$ such that
\begin{align}\label{cctain}
B_Q\subset E_Q\subset 12\sqrt{N} B_Q.
\end{align}
We have
\begin{align}
  {\rm III}_{\sigma,\tilde{\sigma},3}^p
  &\lesssim \sum_{\nu=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ P_Q\in\mathbb{D}_{k-k_0}^\nu}}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{P_Q^{[s_Q]}}-(b)_{E_Q}|^p\nonumber\\
  &\hspace{1.0cm}+\sum_{\tau=1}^{2^N}\sum_{k\in\mathbb{Z}}\sum_{\substack{Q\in\sigma(\mathbb{D}_{\Gamma,k+10})\nonumber\\ A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|\\ W_Q\in\mathbb{D}_{k-k_0}^\tau }}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}|(b)_{W_Q^{[s_Q]}}-(b)_{E_Q}|^p\nonumber\\
  &=:{\rm III}_{\sigma,\tilde{\sigma},31}^p+{\rm III}_{\sigma,\tilde{\sigma},32}^p.
\end{align}
Next, we combine \eqref{cctain} with \eqref{claiminclusion} and the doubling inequality \eqref{doublinginequality} to see that
$$w(P_Q^{[s_Q]})\sim w(E_Q).$$
By Lemma \ref{constructHPQ}, for each $Q\in\sigma(\mathbb{D}_{\Gamma})$ with $P_Q\in\mathbb{D}_{k-k_0}^\nu$, there is a $w$-measurable function $H_{P_Q^{[s_Q]},E_Q}$ on $\mathbb{R}^N$ (unlike the function $H_{Q,m}$ used in estimating ${\rm III}_{\sigma,\tilde{\sigma},1}$ and ${\rm III}_{\sigma,\tilde{\sigma},2}$, $H_{P_Q^{[s_Q]},E_Q}$ cannot be chosen from an arbitrary Haar basis) satisfying
\begin{enumerate}
  \item $|H_{P_Q^{[s_Q]},E_Q}(x)|\lesssim w(E_Q)^{-1/2}\chi_{E_Q}(x)$;\smallskip
  \item $\int_{E_Q} H_{P_Q^{[s_Q]},E_Q}(x)dw(x)=0$;\smallskip
  \item $|(b)_{P_Q^{[s_Q]}}-(b)_{E_Q}|\approx w(E_Q)^{-1/2}|\langle b,H_{P_Q^{[s_Q]},E_Q}\rangle_{L^2(\mathbb{R}^N,dw)}|$.
\end{enumerate}
For simplicity, we denote by
$$\Upsilon^\nu:=\left\{(k,Q):k\in\mathbb{Z}, Q\in\mathcal{\sigma}(\mathbb{D}_{\Gamma,k+10}),A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|,P_Q\in\mathbb{D}_{k-k_0}^\nu\right\}.$$

Therefore,
\begin{align}\label{substitu1}
  {\rm III}_{\sigma,\tilde{\sigma},31}^p\lesssim \sum_{\nu=1}^{2^N}\sum_{(k,Q)\in \Upsilon^\nu}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}w(E_Q)^{-p/2}|\langle b,H_{P_Q^{[s_Q]},E_Q}\rangle_{L^2(\mathbb{R}^N,dw)}|^{p}.
\end{align}
By Lemma \ref{NWOlemma}, there exist NWO sequences $\{\xi_{E_Q}^{(s)}\}$ and $\{\eta_{E_Q}^{(s)}\}$, $s=1,2,3,4$, such that
\begin{align*}
w(E_Q)^{-1/2}|\langle b,H_{P_Q^{[s_Q]},E_Q}\rangle_{L^2(\mathbb{R}^N,dw)}|\lesssim \sum_{s=1}^4\big|\big\langle [b,R_{\D,j}]\xi_{E_Q}^{(s)},\eta_{E_Q}^{(s)}\big\rangle_{L^2(\mathbb{R}^N,dw)}\big|.
\end{align*}
Substituting this inequality into \eqref{substitu1} and then repeating the duality argument as in \eqref{repeatduality}, we deduce that
\begin{align}\label{beclaimm}
{\rm III}_{\sigma,\tilde{\sigma},31}
&\lesssim \sum_{s=1}^4\sum_{\nu=1}^{2^N}\Bigg(\sum_{(k,Q)\in \Upsilon^\nu}2^{-kp}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-p}w(E_Q)^{-p/2}\big|\big\langle [b,R_{\D,j}]\xi_{E_Q}^{(s)},\eta_{E_Q}^{(s)}\big\rangle_{L^2(\mathbb{R}^N,dw)}\big|^{p}\Bigg)^{1/p}\nonumber\\
&\lesssim\sum_{s=1}^4\sum_{\nu=1}^{2^N}\sup_{\|\{\psi_{k,Q}\}_{(k,Q)\in\Upsilon^\nu}\|_{\ell^{p'}}=1}\Bigg\|\sum_{(k,Q)\in\Upsilon^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\psi_{k,Q}(\xi_{E_Q}^{(s)}\otimes \eta_{E_Q}^{(s)})\Bigg\|_{S^{p'}}\|[b,R_{\D,j}]\|_{S^p}.
\end{align}
We {\it claim} that
\begin{align}\label{supinequality2}
\sup_{\|\{\psi_{k,Q}\}_{(k,Q)\in\Upsilon^\nu}\|_{\ell^{p'}}=1}\Bigg\|\sum_{(k,Q)\in\Upsilon^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\psi_{k,Q}(\xi_{E_Q}^{(s)}\otimes \eta_{E_Q}^{(s)})\Bigg\|_{S^{p'}}\lesssim 1.
\end{align}
To show \eqref{supinequality2}, we first write
\begin{align}\label{rhs7777}
&\sum_{(k,Q)\in\Upsilon^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\psi_{k,Q}(\xi_{E_Q}^{(s)}\otimes \eta_{E_Q}^{(s)})\nonumber\\
&=\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{Q\in U_{m,k}^\nu}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\psi_{k,Q} (\xi_{E_Q}^{(s)}\otimes \eta_{E_Q}^{(s)})\nonumber\\
&=\sum_{\tau=1}^{2^N}\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{Q\in U_{m,k}^{\nu,\tau}}2^{-k}\|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|^{-1}\psi_{k,Q} (\xi_{E_Q}^{(s)}\otimes \eta_{E_Q}^{(s)}),
\end{align}
where
$$U_{m,k}^\nu:=\left\{Q\in\sigma(\mathbb{D}_{\Gamma,k+10}):A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|,\ \ell(E_Q)=2^{-(k-m+10)},\ P_Q\in \mathbb{D}_{k-k_0}^\nu\right\},$$
$$U_{m,k}^{\nu,\tau}:=\left\{Q\in\sigma(\mathbb{D}_{\Gamma,k+10}):A_22^{-k+1}\leq \|c_Q-\widetilde{\sigma}(\sigma(c_Q))\|,\ E_Q\in\mathbb{D}^\tau_{k-m+10},\ P_Q\in \mathbb{D}_{k-k_0}^\nu\right\}.$$
Note that for any $Q\in\mathbb{D}^\tau_{k-m+10}$, there are at most $c2^{mN}$ cubes $W\in\sigma(\mathbb{D}_{\Gamma,k+10})$ such that $E_W=Q$, where $c$ is an absolute constant. Denote these cubes by $Q_{\langle m \rangle}^1, Q_{\langle m\rangle}^2,\ldots,Q_{\langle m\rangle}^\varsigma$ with $\varsigma\leq c2^{mN}$. Let $\tilde{\delta}_{Q,m,\ell}=1$ if $E_{Q_{\langle m\rangle}^\ell}=Q$ and $\tilde{\delta}:=0$ otherwise. Let
$$\Delta_{m,k,l}^{\nu,\tau}:=\left\{Q\in \mathbb{D}_{k-m+10}^\tau: A_22^{-k+1}\leq \|c_{Q_{\langle m\rangle}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{\langle m\rangle}^\ell}))\|,\ P_{Q_{\langle m\rangle}^\ell}\in \mathbb{D}_{k-k_0}^\nu \right\}.$$
We have
\begin{align}\label{rhs8888}
&{\rm RHS}\ {\rm of}\ \eqref{rhs7777}\nonumber\\
&=\sum_{\tau=1}^{2^N}\sum_{m\geq 1}\sum_{k\in\mathbb{Z}}\sum_{\ell=1}^{c2^{mN}}\sum_{Q\in\Delta_{m,k,\ell}^{\nu,\tau}}2^{-k}\|c_{Q_{\langle m\rangle}^\ell}-\widetilde{\sigma}(\sigma(c_{Q_{\langle m\rangle}^\ell}))\|^{-1}\tilde{\delta}_{Q,m,\ell}\psi_{k,Q_{\langle m \rangle}^\ell}(\xi_{Q}^{(s)}\otimes \eta_{Q}^{(s)})\nonumber\\

\begin{definition}
Let $U\subset\mathbb{R}^N$ be a convex set.
We say that a non-negative function $f$ is \emph{log-concave on $U$} if for all $x,y\in U$ and all $t\in(0,1)$,
\[
f((1-t)x+ty)\ \ge\ f(x)^{1-t}\,f(y)^t.
\]
\end{definition}
\begin{nota}
Define the projection $\operatorname{proj}_{\mathbb R^{N_0}}:\mathbb R^{N_0}\times\mathbb R^{N-N_0}\to\mathbb R^{N_0}$ by
\[
\operatorname{proj}_{\mathbb R^{N_0}}(u,z):=u .
\]
\end{nota}
\begin{lemma}\label{lem:logconcave-N}
Let $\phi$ be the function as in \eqref{pushforward-density}.
The function $\xi\mapsto \phi(\operatorname{proj}_{\mathbb R^{N_0}}(\xi))$ is log-concave on $\Omega$.
\end{lemma}

\begin{proof}
Fix $a\in \mathbb{R}_+^{N_0}\setminus\{0\}$ and set $L(u):=a\cdot u$. On the open set $\{L>0\}$,
\[
\nabla \log L(u)=\frac{a}{a\cdot u},
\qquad
\nabla^2\log L(u)=-\frac{a\,a^{\mathsf T}}{(a\cdot u)^2},
\]
which is negative semidefinite. Hence $\log L$ is concave on $\{L>0\}$, and therefore $L$ is log-concave on
$\{L>0\}$. Fix $u_1,u_2\in \mathbb{R}_+^{N_0}$ and $t\in(0,1)$. For $\varepsilon>0$ set
\[
u_1^\varepsilon:=u_1+\varepsilon a,
\qquad
u_2^\varepsilon:=u_2+\varepsilon a.
\]
Then $u_1^\varepsilon,u_2^\varepsilon\in\mathbb{R}_+^{N_0}$ and
\[
L(u_i^\varepsilon)=a\cdot u_i+\varepsilon\,a\cdot a
= L(u_i)+\varepsilon\|a\|^2 \ >\ 0,
\qquad i=1,2,
\]
so $u_1^\varepsilon,u_2^\varepsilon\in\{L>0\}$. Since $L$ is log-concave on $\{L>0\}$, we have
\[
L\big((1-t)u_1^\varepsilon+t u_2^\varepsilon\big)\ \ge\ L(u_1^\varepsilon)^{1-t}\,L(u_2^\varepsilon)^t.
\]
Using the linearity of $L$,
\[
L\big((1-t)u_1^\varepsilon+t u_2^\varepsilon\big)
= L\big((1-t)u_1+t u_2\big)+\varepsilon\|a\|^2,
\qquad
L(u_i^\varepsilon)=L(u_i)+\varepsilon\|a\|^2.
\]
Letting $\varepsilon\rightarrow 0$ and using the continuity of $L$ and of the map
$s\mapsto s^\vartheta$ on $[0,\infty)$ for every $\vartheta>0$, we obtain
\[
L\big((1-t)u_1+t u_2\big)\ \ge\ L(u_1)^{1-t}\,L(u_2)^t.
\]
Thus $L$ is log-concave on all of $\mathbb{R}_+^{N_0}$.


For each $v\in R_+$, we apply the above property with $a=a_v:=(a_{v,1},\dots,a_{v,N_{0}})\in\mathbb{R}_+^{N_{0}}\setminus\{0\}$
and $L_v(u):=\sum_{i=1}^{N_{0}}a_{v,i}u_i=a_v\cdot u$ to conclude that each $L_v$ is log-concave on
$\mathbb{R}_+^{\,N_{0}}$. Since
\[
\log\phi(u)=-\log|\det(\Phi)|+2\sum_{v\in R_+}k(v) \log L_v(u)
\]
is the sum of a constant and a
nonnegative linear combination of concave functions, $\phi$ is log-concave on $\mathbb{R}_+^{N_{0}}$.

Finally, note that for $\xi_1=(u_1,z_1)\in\Omega$ and $\xi_2=(u_2,z_2)\in\Omega$ with $u_1,u_2\in\mathbb{R}_+^{N_0}$ and $z_1,z_2\in\mathbb{R}^{N-N_0}$,
\[
\phi(\operatorname{proj}_{\mathbb R^{N_0}}((1-t)\xi_1+t\xi_2))=\phi((1-t)u_1+tu_2)\ge \phi(u_1)^{1-t}\phi(u_2)^t=\phi(\operatorname{proj}_{\mathbb R^{N_0}}(\xi_1))^{1-t}\phi(\operatorname{proj}_{\mathbb R^{N_0}}(\xi_2))^{t}.
\]
so $\xi\mapsto \phi(\operatorname{proj}_{\mathbb R^{N_0}}(\xi))$ is log-concave on $\Omega$. The proof of Lemma \ref{lem:logconcave-N} is complete.
\end{proof}

\begin{definition}\label{def:induced-measure-N}
Let $f:\mathbb{R}^N\to[0,\infty)$ be Lebesgue measurable with
$0<\int_{\mathbb{R}^N} f(x)dx<\infty$. The \emph{induced measure} $\pi_f$ is defined as the probability measure
\[
\pi_f(A):=\frac{\int_A f(x)\,dx}{\int_{\mathbb{R}^N} f(x)\,dx},\quad \text{for all measurable set }  A\subset\mathbb{R}^N.
\]
\end{definition}

\begin{proposition}{\rm \cite[Theorem 4.2]{VempalaSurvey}}\label{prop:KLS-N}
Let $f$ be a log-concave function whose support has diameter $D$, and let $\pi_f$ be the induced measure.
Then for any partition of $\mathbb{R}^N$ into measurable sets $S_1,S_2,S_3$,
\[
\pi_f(S_3)\ \ge\ \frac{2\,{\rm dist}(S_1,S_2)}{D}\,\min\{\pi_f(S_1),\pi_f(S_2)\}.
\]
\end{proposition}

\begin{definition}\label{def:minkowski-N}
Let $(X,d,v)$ be a metric measure space. For a Borel set $A\subset X$ and $\varepsilon>0$, the closed $\varepsilon$-neighbourhood of $A$ is defined as
\[A_\varepsilon:=\{x\in X:\ {\rm dist}(x,A)\leq\varepsilon\}.\]
The codimension Minkowski content of $A\subset X$ is defined as
\[
v^{+}(A):=\liminf_{\varepsilon\rightarrow 0}\frac{v(A_\varepsilon\backslash A)}{\varepsilon}.
\]
\end{definition}

The following lemma establishes a Cheeger-type inequality.
\begin{lemma}\label{lem:cheeger-N}
Let $K\subset\mathbb{R}^N$ be a convex body (i.e.\ a compact convex set with nonempty interior) with diameter $D:={\rm diam}(K)$.
Let $\mathbb{P}$ be a Borel probability measure on $K$ of the form $d\mathbb{P}=\rho\,dx$, where $dx$ is $N$-dimensional
Lebesgue measure and $\rho$ is log-concave on $K$. Then for every measurable $A\subset K$,
\[
\mathbb{P}^+(A)\ \ge\ \frac{2}{D}\,\min\{\mathbb{P}(A),\,1-\mathbb{P}(A)\}.
\]
\end{lemma}

\begin{proof}
Define $f:\mathbb R^N\to[0,\infty)$ by $f=\rho$ on $K$ and $f=0$ on $\mathbb R^N\setminus K$.
Since $K$ is convex and $\rho$ is log-concave on $K$,  $f$
is log-concave on $\mathbb R^N$. Moreover ${\rm supp}(f)\subset\overline K$, hence
${\rm diam}({\rm supp}(f))\le D$.
Since $\int_{\mathbb R^N}f\,dx=\int_K\rho\,dx=1$, we have $\pi_f=\mathbb{P}$ (as measures on $\mathbb R^N$ supported on $K$).
Fix $\varepsilon>0$ and let
$
A_\varepsilon:=\{x\in K:\ {\rm dist}(x,A)\le \varepsilon\}.
$
Consider the measurable partition of $\mathbb R^N$ given by
\[
S_1:=A,\qquad S_2:=K\setminus A_\varepsilon,\qquad S_3:=(A_\varepsilon\setminus A)\cup(\mathbb R^N\setminus K).
\]
Then ${\rm dist}(S_1,S_2)\ge \varepsilon$.
Applying Proposition~\ref{prop:KLS-N} to this partition and using ${\rm diam}({\rm supp}(f))\le D$ gives
\[
\mathbb{P}(A_\varepsilon)-\mathbb{P}(A)=\pi_f(S_3)
\ \ge\ \frac{2\varepsilon}{D}\,\min\{\pi_f(S_1),\pi_f(S_2)\}
=\frac{2\varepsilon}{D}\,\min\{\mathbb{P}(A),1-\mathbb{P}(A_\varepsilon)\}.
\]
Divide by $\varepsilon$:
\[
\frac{\mathbb{P}(A_\varepsilon)-\mathbb{P}(A)}{\varepsilon}
\ \ge\ \frac{2}{D}\,\min\{\mathbb{P}(A),1-\mathbb{P}(A_\varepsilon)\}.
\]
Now as $\varepsilon\downarrow 0$, the sets $A_\varepsilon$ decrease to $\overline A\cap K$. Hence,
$\mathbb{P}(A_\varepsilon)\downarrow \mathbb{P}(\overline A\cap K)$.
If $\mathbb{P}(\overline A\cap K)>\mathbb{P}(A)$ then $\mathbb{P}(A_\varepsilon)-\mathbb{P}(A)\ge c>0$ for small $\varepsilon$,
so $\mathbb{P}^+(A)=+\infty$ and the inequality is trivial.
Otherwise $\mathbb{P}(\overline A\cap K)=\mathbb{P}(A)$, so $1-\mathbb{P}(A_\varepsilon)\rightarrow 1-\mathbb{P}(A)$.
Taking ${\displaystyle\liminf_{\varepsilon\rightarrow 0}}$ yields the desired result.
\end{proof}

\begin{definition}\label{lem:median-N}
Let $(K,\mathbb{P})$ be a probability space and $f\in L^1(\mathbb{P})$. We say that $m_f\in\mathbb{R}$ is a median value of $f$ over $K$ with respect to the measure $\mathbb{P}$ if
\[
\mathbb{P}(\{x\in K:f(x)\ge m_f\})\ge \frac12,\qquad \mathbb{P}(\{x\in K:f(x)\le m_f\})\ge \frac12.
\]
\end{definition}
A median value always exists, but may not be unique (see for example \cite{Journe}). In what follows, we let $m_f$ be an arbitrary median value of $f$.

\begin{lemma}\label{lem:L1Poincare-N}
Let $K\subset\mathbb{R}^N$ be a convex body (i.e.\ a compact convex set with nonempty interior) with diameter $D:={\rm diam}(K)$.
Let $\mathbb{P}$ be a Borel probability measure on $K$ of the form $d\mathbb{P}=\rho(x)\,dx$, where $dx$ is $N$-dimensional
Lebesgue measure and $\rho$ is a log-concave function on $K$. Let $f\in {\rm Lip}(\overline K)$ and let $m_f$ be a median value of $f$.
Then
\[
\int_K |f(x)-m_f|\,d\mathbb{P}(x) \ \le\ \frac{D}{2}\int_K |\nabla f(x)|\,d\mathbb{P}(x).
\]
\end{lemma}
\begin{proof}
The proof is a direct consequence of the combination of Lemma \ref{lem:cheeger-N} with \cite[Theorem~1.1]{MR1396954}. Indeed, we define the shift-invariant functional
\[
L(h):=\inf_{a\in\R}\int_K |h(x)-a|\,d\mathbb{P}(x).
\]
Note that for any measurable set $A\subset K$, we have $\max\{L(\mathbf 1_A),L(-\mathbf 1_A)\}=\min\{\mathbb{P}(A),1-\mathbb{P}(A)\}.$ This, together with Lemma \ref{lem:cheeger-N}, implies that the condition \cite[(1.8)]{MR1396954} holds. By \cite[Theorem~1.1]{MR1396954},
\begin{equation}\label{eq:BH-applied}
\int_K |\nabla f(x)|\,d\mathbb{P}(x) \ \ge\ \frac{2}{D}\,\inf_{a\in\R}\int_K |f(x)-a|\,d\mathbb{P}(x) =\frac{2}{D}\int_K |f(x) -m_f|\,d\mathbb{P}(x) ,
\end{equation}
where the last equality follows from the definition of median value.
\end{proof}

\begin{definition}
We say that $R'\subset\Omega$ is an axis--parallel rectangle if
\[
R' := I_1\times\cdots\times I_N,
\]
with $I_i=[I_i^-,I_i^+]\subset[0,\infty)$ for $1\le i\le N_{0}$ and $I_i=[I_i^-,I_i^+]\subset\mathbb{R}$ for $i>N_{0}$.
\end{definition}

The following lemma  demonstrates that  $(\Omega,\|\cdot\|,d\mu)$ supports a rectangle-version $(1,1)$--Poincar\'e inequality.
\begin{lemma}\label{lem:reducePoincare}
For every axis--parallel rectangle $R'\subset\Omega$  and every Lipschitz function $f:\Omega\to\mathbb{C}$, we have
\[
\fint_{R'}\fint_{R'} |f(x)-f(y)|\,d\mu(x)\,d\mu(y)
\ \le\ {\rm diam}(R')\,\fint_{R'} |\nabla f(x)|\,d\mu(x).
\]
\end{lemma}

\begin{proof}
By decomposing $f=\Re f+i\Im f$ and then using triangular inequality, it suffices to show the conclusion for Lipschitz function $f:\Omega\to\mathbb{R}$.

Define a probability measure supported on $R'$ by
\[
d\mathbb{P}(x):=\frac{1}{\mu(R')}\,d\mu(x)
=\frac{\phi(\operatorname{proj}_{\mathbb R^{N_0}}(x))}{\mu(R')}\,dx,
\]
where the second equality follows from Lemma \ref{mulemma}.
It is equivalent to prove
\begin{align}\label{equiinequality}
\int_{R'}\int_{R'} |f(x)-f(y)|\,d\mathbb{P}(x)\,d\mathbb{P}(y)
\ \le\ {\rm diam}(R')\int_{R'}|\nabla f(x)|\,d\mathbb{P}(x).
\end{align}


To show the inequality \eqref{equiinequality}, we first apply Lemma~\ref{lem:logconcave-N} to see that the function $\xi\mapsto \phi(\operatorname{proj}_{\mathbb R^{N_0}}(\xi))$ is log-concave on $\Omega$.
Since $R'$ is convex and $R'\subset\Omega$, the function
$
x\mapsto \phi(\operatorname{proj}_{\mathbb R^{N_0}}(x))$
is log-concave on $R'$.
By Lemma~\ref{lem:L1Poincare-N},
\[
\int_{R'}\int_{R'} |f(x)-f(y)|\,d\mathbb{P}(x)\,d\mathbb{P}(y)
\le 2\int_{R'} |f(x)-m_f|\,d\mathbb{P}(x)\leq {\rm diam}(R')\int_{R'}|\nabla f(x)|\,d\mathbb{P}(x),
\]
which shows the desired estimate \eqref{equiinequality}. Thus, the proof of Lemma \ref{lem:reducePoincare} is complete.
\end{proof}

Denote by
$$B_\Omega(x_0,r):=\{x\in\Omega:\ \|x-x_0\|<r\}.$$
The following lemma demonstrates that $(\Omega,\|\cdot\|,d\mu)$ is a doubling metric measure space.
\begin{lemma}\label{lem:ball-2sided-N}
There exists a constant $C\ge 1$ such that
for every $x_0=(u_0,z_0)\in\Omega$ and every $r>0$,
\begin{equation}\label{eq:ball-2sided-Omega}
C^{-1}\,r^N\prod_{v\in R_+}\Big(L_{v,0}+r\Big)^{2k(v)}
\ \le\
\mu\big(B_\Omega(x_0,r)\big)
\ \le\
C\,r^N\prod_{v\in R_+}\Big(L_{v,0}+r\Big)^{2k(v)},
\end{equation}
where
\[
L_v(u):=\sum_{i=1}^{N_{0}} a_{v,i}u_i,
\qquad
L_{v,0}:=L_v(u_0)=\sum_{i=1}^{N_{0}} a_{v,i}(u_0)_i.
\]
In particular, there exists a constant $C_{\mathrm{dbl},\Omega}\geq 1$ such that for every $x\in\Omega$ and every $r>0$,
\[
\mu\big(B_\Omega(x,2r)\big)\ \le\ C_{\mathrm{dbl},\Omega}\;\mu\big(B_\Omega(x,r)\big).
\]
\end{lemma}

\begin{proof}
Fix $x_0=(u_0,z_0)\in\Omega$ and $r>0$. For each $v\in R_+$ define the linear form
\[
L_v(u):=a_v\cdot u,\qquad a_v:=(a_{v,1},\dots,a_{v,N_{0}})\in \mathbb{R}_+^{N_{0}}\setminus\{0\}.
\]
Since $R_+$ is finite and each $a_v\neq 0$, we have
\[
L_*:=\max_{v\in R_+}\|a_v\|<\infty,
\qquad
m_*:=\min_{v\in R_+}\sum_{i=1}^{N_{0}} a_{v,i}>0.
\]

\noindent
\textbf{Upper bound.}
Let $x=(u,z)\in B_\Omega(x_0,r)$. Then
\[
\|u-u_0\|\ \le\ \|x-x_0\|\ \le\ r.
\]
Hence, for every $v\in R_+$,
\[
|L_v(u)-L_{v,0}|
=\big|a_v\cdot (u-u_0)\big|
\le \|a_v\|\cdot\|u-u_0\|
\le L_*\,r.
\]
Since $L_v(u)\ge 0$ on $\Omega$ (all coefficients and coordinates are nonnegative), we obtain
\[
L_v(u)\le L_{v,0}+L_*r\le C_1\,(L_{v,0}+r),
\qquad
C_1:=\max\{1,L_*\}.
\]
Therefore,
\[
\phi(u)
=\prod_{v\in R_+} L_v(u)^{2k(v)}
\le C_1^{2\sum_{v\in R_+}k(v)}\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}
=:C_2\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}.
\]
Integrating over $B_\Omega(x_0,r)$ and using
$|B_\Omega(x_0,r)|\le |B(x_0,r)|=\omega_N r^N$, where $\omega_N$ is the volume of the unit ball in $\mathbb{R}^N$, gives
\[
\mu\big(B_\Omega(x_0,r)\big)
=\int_{B_\Omega(x_0,r)}\phi(u)\,dx
\le C_2\,|B(x_0,r)|\,\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}
\le C_3\,r^N\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)},
\]
with $C_3:=C_2\,\omega_N$.

\medskip\noindent
\textbf{Lower bound.}
We construct a smaller Euclidean ball $B_c$ contained in $B_\Omega(x_0,r)$ on which $\phi$ admits a uniform
positive lower bound in terms of $\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}$. To this end, we first choose
\[
\lambda:=1+\frac{L_*}{m_*}>1,
\qquad
\eta:=\frac{1}{4\big(\sqrt{N_{0}}\,\lambda+1\big)},
\qquad
\theta:=\lambda\,\eta.
\]
These parameters depend only on $L_*$ and $m_*$, and satisfy
\[
\theta> \eta>0,
\qquad
\sqrt{N_{0}}\,\theta+\eta=\eta(\sqrt{N_{0}}\lambda+1)=\frac14<1.
\]
Let $\mathbf{1}_{N_{0}}:=(1,\dots,1)\in\mathbb{R}^{N_{0}}$ and define the center
\[
x_c:=(u_c,z_c):=\big(u_0+\theta r\,\mathbf{1}_{N_{0}},\ z_0\big),
\qquad
B_c:=B(x_c,\eta r).
\]
We {\it claim} that $B_c\subset B_\Omega(x_0,r)$. To see this inclusion, we first apply the triangle inequality to see that for any $x\in B_c$,
\[
\|x-x_0\|\le \|x-x_c\|+\|x_c-x_0\|
\le \eta r+\theta r\sqrt{N_{0}}
\le r,
\]
since $\sqrt{N_{0}}\theta+\eta<1$. Hence $B_c\subset B(x_0,r)$. Moreover, for any $x=(u,z)\in B_c$, we have $\|u-u_c\|\le \|x-x_c\|\le \eta r$. This implies that for each $1\le i\le N_{0}$,
\[
u_i\ge (u_c)_i-\eta r=(u_0)_i+(\theta-\eta)r\ge 0,
\]
since $u_0\in\mathbb{R}_+^{N_{0}}$ and $\theta\ge\eta$. Thus $u\in\mathbb{R}_+^{N_{0}}$ and $x\in\Omega$. Therefore,  $B_c\subset B_\Omega(x_0,r)$. This finishes  the proof of the  claim.

\smallskip
Next we obtain a uniform lower bound for $L_v(u)$ on $B_c$.
Write any $x=(u,z)\in B_c$ as
\[
u=u_0+\theta r\,\mathbf{1}_{N_{0}}+\Delta_u,
\qquad
z=z_0+\Delta_z,
\qquad
\|\Delta_u\|^2+\|\Delta_z\|^2\le (\eta r)^2,
\]
so in particular $\|\Delta_u\|\le \eta r$.
Then for all $x=(u,z)\in B_c$ and all $v\in R_+$,
\begin{align*}
L_v(u)
&=a_v\cdot u
=a_v\cdot u_0+\theta r\,a_v\cdot\mathbf{1}_{N_{0}}+a_v\cdot\Delta_u \\
&\ge L_{v,0}+\theta r\sum_{i=1}^{N_{0}} a_{v,i}-\|a_v\|\,\|\Delta_u\|
\ge L_{v,0}+rK_0\geq c_0(L_{v,0}+r).
\end{align*}
with
\[
K_0:=\theta m_*-\eta L_*=\eta(\lambda m_*-L_*)=\eta m_*>0,\qquad c_0:=\min\{1,K_0\}\in(0,1].
\]
Consequently, for all $u\in B_c$,
\[
\phi(u)
=\prod_{v\in R_+}L_v(u)^{2k(v)}
\ge c_0^{2\sum_{v\in R_+}k(v)}\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}
=:C_4\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}.
\]
This, in combination with the fact that $B_c\subset B_\Omega(x_0,r)$, yields
\[
\mu\big(B_\Omega(x_0,r)\big)
\ge \mu(B_c)
=\int_{B_c}\phi(u)\,dx
\ge C_4\,|B_c|\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}
= C_5\,r^N\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)},
\]
where $C_5:=C_4\,\omega_N\eta^N>0$.

The second statement follows immediately from the first one. Indeed, by the first statement,
\[
\mu\big(B_\Omega(x_0,2r)\big)
\le C_3\,(2r)^N\prod_{v\in R_+}(L_{v,0}+2r)^{2k(v)}
\le C_3\,2^{N+\mathcal{K}}\,r^N\prod_{v\in R_+}(L_{v,0}+r)^{2k(v)}
\le C_3C_5^{-1}\,2^{N+\mathcal{K}}\,\mu\big(B_\Omega(x_0,r)\big),
\]
where we set $\mathcal{K}:=2\sum_{v\in R_+}k(v)$. Therefore, $(\Omega,\|\cdot\|,\mu)$ is doubling with the doubling constant $C_{\mathrm{dbl},\Omega}:=C_3C_5^{-1}\,2^{N+\mathcal{K}}$.
 This finishes the proof of Lemma \ref{lem:ball-2sided-N}.
\end{proof}
Let $\Phi:\mathbb{R}^N\longrightarrow\mathbb{R}^N$ be the linear isomorphism given in  Corollary~\ref{coro:chamber-orthant} such that
$
\Phi(\Gamma)=\Omega.
$
Since $\Phi$ is linear and invertible, there exist positive constants $c_\Phi$ and $C_\Phi$ such that for any $x,y\in\mathbb{R}^N$,
\begin{equation}\label{eq:biLip-N}
c_\Phi\,\|x-y\|\ \le\ \|\Phi(x)-\Phi(y)\|\ \le\ C_\Phi\,\|x-y\|.
\end{equation}
Denote by $B_\Gamma(x,r):=\{y\in\Gamma:\ \|y-x\|<r\}$. It follows from \eqref{eq:biLip-N} that for any $x_0\in\Gamma$ and $r>0$,
\begin{align}\label{beforedoublingball}
B_{\Omega}(\Phi(x_0),c_\Phi r)\subset \Phi(B_{\Gamma}(x_0,r))\subset B_{\Omega}(\Phi(x_0),C_\Phi r).
\end{align}
This, together with Lemma  \ref{lem:ball-2sided-N}, yields
\begin{align}\label{doublingball}
  \mu(\Phi(B_{\Gamma}(x_0,r))) \simeq \mu(B_{\Omega}(\Phi(x_0),C_\Phi r)).
\end{align}
\begin{lemma}\label{construction-rectangle}
  For all $x_0\in\Gamma$ and $r>0$, one can find an axis--parallel rectangle $R'$ on $\Omega$ such that
\begin{align}\label{step3goal}
B_{\Omega}(\Phi(x_0),C_\Phi r)\subset R'\subset B_{\Omega}(\Phi(x_0),\sqrt{N}C_\Phi r).
\end{align}
Furthermore, we have
\begin{align}\label{Rdoubling}
  \mu(R')\simeq \mu(B_{\Omega}(\Phi(x_0),C_\Phi r)).
\end{align}

\end{lemma}
\begin{proof}
We write
$
(u_0,z_0):=\Phi(x_0)\in\Omega.
$
and define the axis--parallel rectangle
\begin{equation}\label{eq:rectangle-choice}
R':=\Big(\prod_{i=1}^{N_{0}} \big[\max\{0,(u_0)_i-C_\Phi r\},\ (u_0)_i+C_\Phi r\big]\Big)
\times
\Big(\prod_{j=1}^{N-N_{0}}\big[(z_0)_j-C_\Phi r,\ (z_0)_j+C_\Phi r\big]\Big).
\end{equation}

On the one hand, it follows immediately that $B_\Omega(\Phi(x_0),C_\Phi r)\subset R'$. On the other hand, since $R'$ is a rectangle with half-side lengths bounded by $C_\Phi r$ in each coordinate, we have
\begin{align*}
R'\subset B_\Omega(\Phi(x_0),\sqrt{N}\,C_\Phi r).
\end{align*}
The second statement follows from the first one together with Lemma \ref{lem:ball-2sided-N}. The proof of Lemma \ref{construction-rectangle} is finished.
\end{proof}
The following theorem demonstrates that $(\Gamma,\|\cdot\|,dw)$ is a doubling metric measure space.


\begin{theorem}\label{main:dunkldoubling}
There exists a constant $C\ge 1$ such that
for every $x\in\Gamma$ and every $r>0$,
\begin{equation}\label{eq:ball-2sided-Gamma}
C^{-1}\,r^N \prod_{v \in R_+}(|\langle x, v \rangle |+r)^{2k(v)}
\ \le\
w\big(B_\Gamma(x,r)\big)
\ \le\
C\,r^N \prod_{v \in R_+}(|\langle x, v \rangle |+r)^{2k(v)}.
\end{equation}
In particular, there exists a constant $C_{\mathrm{dbl},\Gamma}\geq 1$ such that for every $x\in\Gamma$ and every $r>0$,
\[
w\big(B_\Gamma(x,2r)\big)\ \le\ C_{\mathrm{dbl},\Gamma}\;w\big(B_\Gamma(x,r)\big).
\]
Moreover, for all $x\in\Gamma$ and $r>0$, we have
\begin{align}\label{globallocalvol}
w(B_\Gamma(x,r))\sim w(B(x,r)).
\end{align}
\end{theorem}
\begin{proof}
The main strategy of the proof is to transfer the ball estimates from $(\Omega,\|\cdot\|,\mu)$ back to $(\Gamma,\|\cdot\|,dw)$. Fix $x\in\Gamma$ and $r>0$. It follows from the equality \eqref{defmu} that
\begin{equation}\label{eq:pushforward-ball}
w\big(B_\Gamma(x,r)\big)=\mu\big(\Phi(B_\Gamma(x,r))\big).
\end{equation}
Applying $\mu$ to the inclusions \eqref{beforedoublingball} and using \eqref{eq:pushforward-ball} yields
\begin{equation}\label{eq:wball-between-muballs}
\mu\big(B_{\Omega}(\Phi(x),c_\Phi r)\big)
\ \le\
w\big(B_\Gamma(x,r)\big)
\ \le\
\mu\big(B_{\Omega}(\Phi(x),C_\Phi r)\big).
\end{equation}
Write $\Phi(x)=(u,z)\in\Omega$. For each $v\in R_+$ recall the linear form
$L_{v,0}:=\sum_{i=1}^{N_0}a_{v,i}u_i$.
Combining the inequality \eqref{eq:wball-between-muballs} with Lemma~\ref{lem:ball-2sided-N} and the equality \eqref{changevariable}, we obtain that
\begin{equation}\label{eq:wball-in-terms-of-L}
w\big(B_\Gamma(x,r)\big)\ \simeq\ r^N\prod_{v\in R_+}\big(L_{v,0}+r\big)^{2k(v)}=r^N\prod_{v\in R_+}\big(|\langle x,v\rangle|+r\big)^{2k(v)}.
\end{equation}
This proves \eqref{eq:ball-2sided-Gamma}. The second statement follows immediately from \eqref{eq:ball-2sided-Gamma}. The third statement follows from the inequalities \eqref{eq:ball-2sided-Gamma} and \eqref{twosideinequ}. This completes the proof of Theorem~\ref{main:dunkldoubling}.
\end{proof}

The following theorem demonstrates that the metric measure space $(\Gamma,\|\cdot\|,dw)$ supports $(1,1)$--Poincar\'e inequality.

{\color{red}[Zhijie will try to extends the $(1,1)$--Poincar\'e inequality to something like $(p,q)$--Poincar\'e inequality with some minor modifications, due to its independent interest in harmonic analysis]}
\begin{theorem}\label{main:Poincaredunkl}
There exist constants $\lambda\ge1$ and $C>0$ such that for every $x_0\in\Gamma$, $r>0$, and every Lipschitz function $f:\Gamma\to\mathbb{C}$,
\begin{equation}\label{eq:Poincare11-chamber}
\fint_{B_\Gamma(x_0,r)}
\big|f(x)-\fint_{B_\Gamma(x_0,r)}f(y)dw(y)\big|\,dw(x)
\;\le\;
C\,r\,\fint_{B_\Gamma(x_0,\lambda r)} |\nabla f(z)|\,dw(z).
\end{equation}
%{\color{red}where $|\nabla f|$ denotes the Euclidean length of the gradient of $f$ (defined almost everywhere for Lipschitz $f$ and serving as an upper gradient).}
%%where $\nabla f$ denotes the a.e.\ gradient of a Lipschitz extension of $f$ to $\mathbb R^N$.
\end{theorem}
\begin{proof}
By decomposing $f=\Re f+i\Im f$ and then using triangular inequality, it suffices to show that for every $x_0\in\Gamma$, $r>0$, and every Lipschitz function $f:\Gamma\to\mathbb{R}$,
\begin{equation}\label{eq:Poincare11-chamber}
\fint_{B_\Gamma(x_0,r)}\fint_{B_\Gamma(x_0,r)}
\big|f(x)-f(y)\big|\,dw(x)\,dw(y)
\;\le\;
C\,r\,\fint_{B_\Gamma(x_0,\lambda r)} |\nabla f(z)|\,dw(z).
\end{equation}
To show this inequality, we first let $g:=f\circ\Phi^{-1}$ on $\Omega$. It follows that $g\in{\rm Lip}(\Omega)$ with
\begin{equation}\label{eq:grad-comp-S2}
|\nabla g(x)|
 \le
\|D\Phi^{-1}\|_{\mathrm{op}}\;|\nabla f(\Phi^{-1}x)|
\qquad\text{for}\ x\in\Omega.
\end{equation}
For all $x_0\in\Gamma$ and $r>0$, we let $R'$ be the axis--parallel rectangle given in Lemma \ref{construction-rectangle}. Then
\begin{align}\label{plugterm}
&\fint_{B_\Gamma(x_0,r)}\fint_{B_\Gamma(x_0,r)}|f(x)-f(y)|\,dw(x)\,dw(y)\nonumber\\
(\text{By change of variables})&=
\fint_{\Phi(B_\Gamma(x_0,r))}\fint_{\Phi(B_\Gamma(x_0,r))}|g(x)-g(y)|\,d\mu(x)\,d\mu(y)\nonumber\\
(\text{By}\ \eqref{beforedoublingball}\ \text{and}\ \eqref{doublingball})&\lesssim
\fint_{B_{\Omega}(\Phi(x_0),C_\Phi r)}\fint_{B_{\Omega}(\Phi(x_0),C_\Phi r)}|g(x)-g(y)|\,d\mu(x)\,d\mu(y)\nonumber\\
(\text{By}\ \eqref{step3goal}\ \text{and}\ \eqref{Rdoubling})&\lesssim
\fint_{R'}\fint_{R'}|g(x)-g(y)|\,d\mu(x)\,d\mu(y)\nonumber\\
(\text{By Lemma}\ \ref{lem:reducePoincare})&\lesssim {\rm diam}(R')\,\fint_{R'} |\nabla g(x)|\,d\mu(x)\nonumber\\
(\text{By}\ \eqref{eq:grad-comp-S2})&\lesssim {\rm diam}(R')\,
\fint_{\Phi^{-1}(R')} |\nabla f(x)|\,dw(x)\nonumber\\
(\text{By}\  \eqref{step3goal})&\lesssim r\,
\fint_{\Phi^{-1}(R')} |\nabla f(x)|\,dw(x).
\end{align}




To continue, we apply the inclusion \eqref{step3goal}  to see that for every $\xi\in R'$ we have $\|\xi-\Phi(x_0)\|\le \sqrt{N}C_\Phi r$.
Combining with the inequality \eqref{eq:biLip-N}, we deduce that
\[
\|\Phi^{-1}(\xi)-x_0\|
\le \frac{1}{c_\Phi}\,\|\xi-\Phi(x_0)\|
\le \frac{\sqrt{N}C_\Phi}{c_\Phi}\,r.
\]
Since $R'\subset\Omega$ implies $\Phi^{-1}(R')\subset\Gamma$, we see that
\begin{equation}\label{eq:preimage-in-ball}
\Phi^{-1}(R')\ \subset\ B_\Gamma(x_0,\lambda r),
\qquad \text{where}\
\lambda:=\frac{\sqrt{N}C_\Phi}{c_\Phi}\ \ge 1.
\end{equation}
Moreover, it follows from \eqref{beforedoublingball} and \eqref{step3goal} that $B_{\Gamma}(x_0,r)\subset \Phi^{-1}(R')$. Therefore,
\begin{align*}
\fint_{\Phi^{-1}(R')}|\nabla f(x)|\,dw(x)
&=\frac{1}{w(\Phi^{-1}(R'))}\int_{\Phi^{-1}(R')}|\nabla f(x)|\,dw(x)\\
(\text{By}\  \eqref{step3goal}\ \text{and}\ \eqref{eq:preimage-in-ball})&\le \frac{1}{w(B_\Gamma(x_0,r))}\int_{B_\Gamma(x_0,\lambda r)}|\nabla f(x)|\,dw(x)\\
(\text{By Theorem} \ \ref{main:dunkldoubling})&\lesssim \frac{1}{w(B_\Gamma(x_0,\lambda r))}\int_{B_\Gamma(x_0,\lambda r)}|\nabla f(x)|\,dw(x).
\end{align*}
Plugging this inequality into \eqref{plugterm} yields
\[
\fint_{B_\Gamma(x_0,r)}\fint_{B_\Gamma(x_0,r)}
|f(x)-f(y)|\,dw(x)\,dw(y)
\lesssim\,r\,\fint_{B_\Gamma(x_0,\lambda r)}|\nabla f(x)|\,dw(x),
\]
This completes the proof of Theorem \ref{main:Poincaredunkl}.
\end{proof}


\section{Failure of global Poincar\'e inequality}
\setcounter{equation}{0}

{\color{red}[Comments: it seems better to extend the following result to general $N$ and general Weyl group]}
\begin{proposition}\label{lem:counterexample-Poincare}
Let $k>0$ and $w_k$ be the Dunkl measure on $\mathbb{R}$ given by $dw_k(x)=|x|^{2k}\,dx$. Then there do not exist constants $C\ge1$ and $\lambda\ge1$ such that the following inequality holds for all intervals $I\subset\mathbb{R}$ and all Lipschitz functions $f$:
\begin{equation}\label{eq:Poincare-claim}
\fint_{I}\fint_{I} |f(x)-f(y)|\,dw_k(x)\,dw_k(y)
\;\le\; C\,|I|\,\fint_{\lambda I} |f'(z)|\,dw_k(z).
\end{equation}
\end{proposition}

\begin{proof}
Fix $r>0$ and let $I=(-r,r)$, so that by evenness of the weight we have
\[
w_k(I_-)=w_k(I_+)=\tfrac12\,w_k(I),
\qquad I_-:=(-r,0),\ I_+:=(0,r).
\]
For $M\in\mathbb{N}$, define $f_M(x):=\tanh(Mx)$. It follows that  $f_M$ is smooth, $|f_M|\le1$, and $f_M(x)\to\operatorname{sgn}(x)$, as $M\rightarrow \infty$,  for every $x\neq0$.


On the one hand, by the dominated convergence theorem,
\[
\lim_{M\to\infty}\ \fint_I\fint_I |f_M(x)-f_M(y)|\,dw_k(x)\,dw_k(y)
=\fint_I\fint_I |\operatorname{sgn}(x)-\operatorname{sgn}(y)|\,dw_k(x)\,dw_k(y).
\]
Since $|\operatorname{sgn}(x)-\operatorname{sgn}(y)|=0$ if $xy>0$ and $|\operatorname{sgn}(x)-\operatorname{sgn}(y)|=2$ if $xy<0$, we have
\[
\fint_I\fint_I |\operatorname{sgn}(x)-\operatorname{sgn}(y)|\,dw_k(x)\,dw_k(y)
=\frac{4\,w_k(I_-)w_k(I_+)}{w_k(I)^2}
=\frac{4\cdot(w_k(I)/2)\cdot(w_k(I)/2)}{w_k(I)^2}=1.
\]
Therefore,
\begin{equation}\label{eq:LHS-limit}
\lim_{M\to\infty}\ \fint_I\fint_I |f_M(x)-f_M(y)|\,dw_k(x)\,dw_k(y)=1.
\end{equation}
On the other hand, we apply the change of variables to see that
\[
\int_{\lambda I} |f_M'(z)|\,dw_k(z)
=\int_{-\lambda r}^{\lambda r} M\operatorname{sech}^2(Mz)\,|z|^{2k}\,dz
=M^{-2k}\int_{-\lambda Mr}^{\lambda Mr} \operatorname{sech}^2(u)\,|u|^{2k}\,du.
\]
Since $k>0$ and $\int_{\mathbb{R}}\operatorname{sech}^2(u)\,|u|^{2k}\,du<\infty$, we have
\[
\int_{\lambda I} |f_M'(z)|\,dw_k(z)\lesssim M^{-2k}.
\]
Dividing by the fixed positive number $w_k(\lambda I)$ yields
\begin{equation}\label{eq:RHS-limit}
|I|\,\fint_{\lambda I} |f_M'(z)|\,dw_k(z)
=\frac{|I|}{w_k(\lambda I)}\int_{\lambda I}|f_M'(z)|\,dw_k(z)\;\longrightarrow\;0\quad
\text{as}\ M\to\infty.
\end{equation}

\smallskip
Combining \eqref{eq:LHS-limit} and \eqref{eq:RHS-limit}, we see that the inequality \eqref{eq:Poincare-claim} fails for the sequence $f_M$ as $M\to\infty$, no matter how large one chooses the constants $C\ge1$ and $\lambda\ge1$. This contradiction proves that no universal constants $C,\lambda$ can make \eqref{eq:Poincare-claim} valid for all intervals and all Lipschitz $f$.
\end{proof}

\section{Constant characterization}

In this section, we provide a proof for Theorem \ref{constantchar}. The sufficient condition is obvious, since $[b,R_{\D,j}]=0$ when $b$ is a constant. Thus, it suffices to show the necessary condition.
 It suffices to consider  the critical case $p=N$, by the inclusion $ S^p\subset S^{q}$ for $p<q$.

\begin{nota}
Let $\Gamma\subset \mathbb{R}^N$ be a Weyl chamber so that (by Lemma \ref{lem:simple_system_associated_to_chamber})
\begin{align*}
\Gamma=\Bigl\{x\in\mathbb{R}^{N}:\ \langle x,\beta\rangle\ge 0\ \text{for all}\  \beta\in\Delta\Bigr\},
\end{align*}
where $\Delta=\{\alpha_{1},\ldots,\alpha_{N_{0}}\}$ is the simple system associated with $\Gamma$.
For any $\alpha\in\Delta$, let
\[
F_\alpha:=\Bigl\{x\in\mathbb{R}^N:\ \langle x,\alpha\rangle=0,\ \langle x,\beta\rangle\ge 0\ \text{for all}\ \beta\in\Delta\setminus\{\alpha\}\Bigr\}

be the facet of $\Gamma$ associated with $\alpha$.
Moreover, let
\[
\mathrm{relint}(F_\alpha)
:=\Bigl\{x\in\mathbb{R}^N:\ \langle x,\alpha\rangle=0,\ \langle x,\beta\rangle>0\ \text{for all}\ \beta\in\Delta\setminus\{\alpha\}\Bigr\}
\]
be the relative interior of $F_\alpha$ (taken in its affine hull).
\end{nota}

 \begin{lemma}\label{nonempty}
Let $\Gamma\subset \mathbb{R}^N$ be a Weyl chamber. For any $\alpha\in\Delta$, we have
$$\operatorname{relint}(F_\alpha)\neq \varnothing.$$
\end{lemma}
\begin{proof}
Write $\alpha=\alpha_i$ for some $i\in\{1,\dots,N_0\}$.
Let $G=(\langle\alpha_j,\alpha_k\rangle)_{j,k=1}^{N_0}$ be the Gram matrix, which is invertible since $\Delta$ is a basis of $W$ (see Lemma \ref{lem:simple_system_associated_to_chamber}). Let $\varrho\in\mathbb{R}^{N_0}$ be defined by $\varrho_i=0$ and $\varrho_j=1$ for $j\neq i$. Set $\upsilon=G^{-1}\varrho$. Define $z:=\sum_{j=1}^{N_0} \upsilon_j\alpha_j\in W\subset\mathbb{R}^N$.
Then for each $k=1,\dots,N_0$,
\[
\langle z,\alpha_k\rangle=\sum_{j=1}^{N_0}\upsilon_j\langle\alpha_j,\alpha_k\rangle=(G\upsilon)_k=\varrho_k.
\]
In particular, $\langle z,\alpha\rangle=\langle z,\alpha_i\rangle=\varrho_i=0$ and
$\langle z,\beta\rangle=1>0$ for all $\beta\in\Delta\setminus\{\alpha\}$. Hence, $z\in\mathrm{relint}(F_\alpha)$,
which implies that $\mathrm{relint}(F_\alpha)\neq\varnothing$.
\end{proof}

\frac{|\langle z_\alpha,\varpi\rangle|}{2\|\varpi\|}\ >\ 0.
\]
Note that for any $x\in\mathbb{R}^N$ with $\|x-z_\alpha\|<\delta_\alpha$ and any $\varpi\in R\setminus\{\pm\alpha\}$, by the Cauchy--Schwarz inequality we have
\[
\big|\langle x,\varpi\rangle-\langle z_\alpha,\varpi\rangle\big|
=\big|\langle x-z_\alpha,\varpi\rangle\big|
\le \|x-z_\alpha\|\,\|\varpi\|
<\delta_\alpha\,\|\varpi\|
\le \frac12\,|\langle z_\alpha,\varpi\rangle|,
\]
and therefore, $\langle x,\varpi\rangle$ has the same sign as $\langle z_\alpha,\varpi\rangle$.

In particular, since $z_\alpha\in\mathrm{relint}(F_\alpha)$, we have $\langle z_\alpha,\beta\rangle>0$ for all
$\beta\in\Delta\setminus\{\alpha\}$, and thus $\langle x,\beta\rangle>0$ for all $x\in U_\alpha$. Hence,
\[
\Gamma\cap U_\alpha=\{x\in U_\alpha:\langle x,\alpha\rangle\ge 0\}
=U_\alpha^+\cup(F_\alpha\cap U_\alpha).
\]
Applying $\sigma_\alpha$ to the equality above and using $\sigma_\alpha(U_\alpha)=U_\alpha$, $\sigma_\alpha(U_\alpha^+)=U_\alpha^-$ and
$\sigma_\alpha(F_\alpha)=F_\alpha$, we obtain the second identity in \eqref{Gammaequa}. This completes the proof of Lemma \ref{lem:local_structure_near_facet}.
\end{proof}

\begin{lemma}\label{constructionball}
Let $\Gamma\subset \mathbb{R}^N$ be a Weyl chamber.
Then for each $\alpha\in \Delta$, there exist two families of balls
$\{B_m\}_{m\ge 1}\subset \Gamma$ and $\{\hat{B}_m\}_{m\ge 1}\subset \sigma_\alpha(\Gamma)$
such that each family is pairwise disjoint, and such that for each $m\ge1$,

Moreover, for every $m\ge 1$
and every $(x,y)\in B_m\times \hat{B}_m$,
\begin{align}\label{twosideinequalityrm}
11r_m \le \|x-y\| \le 15r_m.
\end{align}
\end{lemma}

\begin{lemma}{\rm \cite[Corollary 4.3]{HKarxiv2}}\label{Hytonen-KorteLemma}
 Let $(X,\rho,\mu)$ be a doubling metric measure
space having lower dimension $d$ and supporting the $(1,d)$--Poincar\'e inequality. If $f\in L^1_{\mathrm{loc}}(X,\mu)$ satisfies
\begin{equation}
\Bigl[(x,y)\mapsto \frac{|f(x)-f(y)|^d}{\mu(B(x,\rho(x,y)))^2}\Bigr]\in L^1_{\mathrm{loc}}(X\times X,\mu\times\mu),
\end{equation}
then $f$ is equal to a constant almost everywhere on $X$.
\end{lemma}

\begin{proposition}\label{constantpro}
Suppose $N\geq 2$ and $b\in  L^1_{{\rm loc}}(\mathbb R^{N},dw)$. If $[b,R_{\D,j}]\in S^N$ for some $j\in\{1,2,...,N\}$, then $b$ is a constant on $\mathbb{R}^N$ almost everywhere.
\end{proposition}
\begin{proof}
The proof is divided into three steps.

{\bf $\bullet$ Step 1.} {\it Show that $b\equiv c_{\Gamma}$ almost everywhere for some constant $c_\Gamma$ on each Weyl chamber $\Gamma$.}

By Lemma \ref{standard}, if $[b,R_{\D,j}]\in S^N$ for some $j\in\{1,2,...,N\}$, then $b\in B^N(\mathbb{R}^N,dw)$. This, together with the inequality \eqref{globallocalvol}, yields that for every Weyl chamber $\Gamma$, we have
$$\|b\|_{B^N(\Gamma,dw)}:=\left(\int_{\Gamma}\int_{\Gamma}\frac{|b(x)-b(y)|^N}{w(B_\Gamma(x,\|x-y\|))^2}dw(x)dw(y)\right)^{1/N}<+\infty.$$
By Theorem \ref{main:dunkldoubling}, $(\Gamma,\|\cdot\|,dw)$ is a doubling metric measure space having lower dimension $N$. 
Together with Theorem \ref{main:Poincaredunkl} and H\"{o}lder's inequality, this implies that $(\Gamma,\|\cdot\|,dw)$ supports a $(1,N)$--Poincar\'e inequality. It follows from Lemma \ref{Hytonen-KorteLemma} that $b\equiv c_{\Gamma}$ on $\Gamma$ almost everywhere for some constant $c_\Gamma$.

{\bf $\bullet$ Step 2.} {\it Show that $c_{\Gamma}=c_{\sigma_\alpha(\Gamma)}$ for each Weyl chamber $\Gamma$ and $\alpha\in \Delta$.}

Indeed, if this is not the case, then $A:=c_{\Gamma}-c_{\sigma_\alpha(\Gamma)}\neq 0$ for some Weyl chamber $\Gamma$ and some $\alpha\in \Delta$(the simple system associated with $\Gamma$). Let $\{(B_m,\hat{B}_m)\}_{m\geq 1}$ be the family of pairs of balls constructed in Lemma \ref{constructionball}. By \eqref{twosideinequalityrm} and the doubling inequality \eqref{doublinginequality}, for all $(x,y)\in B_m\times \hat{B}_m$,
$$w(B(x,\|x-y\|))\sim w(B_m)\sim w(\hat{B}_m).$$


It follows that
\begin{align*}
I_m:=\int_{B_m}\int_{\hat{B}_m}\frac{|b(x)-b(y)|^N}{w(B(x,\|x-y\|))^2}dw(x)dw(y)\gtrsim A^N.
\end{align*}
Since for any $m \neq n$, $B_m \cap B_n = \varnothing$ and $\hat{B}_m \cap \hat{B}_n = \varnothing$, we have
\begin{align*}
\|b\|_{B^N(\mathbb{R}^N,dw)}^N\geq \sum_{m=1}^\infty I_m \gtrsim \sum_{m=1}^\infty A^N=+\infty.
\end{align*}
This is a contradiction to $b\in B^N(\mathbb{R}^N,dw)$. Therefore, $c_{\Gamma}=c_{\sigma_\alpha(\Gamma)}$.

{\bf $\bullet$ Step 3.} {\it Show that $c_{\Gamma_1}=c_{\Gamma_2}$ for any two distinct Weyl chambers $\Gamma_1$ and $\Gamma_2$.}

Fix two Weyl chambers $\Gamma_1$ and $\Gamma_2$. By Lemma~\ref{lem:gallery-simple}, there exist an integer $M\ge0$
and roots $\alpha_1,\dots,\alpha_M\in R$ such that, if we define $\Gamma^{(0)}:=\Gamma_2$ and
\[
\Gamma^{(k)}:=\sigma_{\alpha_k}\big(\Gamma^{(k-1)}\big),\qquad k=1,\dots,M,
\]
then $\alpha_k\in\Delta(\Gamma^{(k-1)})$ for each $k=1,\dots,M$, and $\Gamma^{(M)}=\Gamma_1$.

Applying Step~2 to the Weyl chamber $\Gamma^{(k-1)}$ and the simple root $\alpha_k\in\Delta(\Gamma^{(k-1)})$,
we obtain
\[
c_{\Gamma^{(k)}}=c_{\sigma_{\alpha_k}(\Gamma^{(k-1)})}=c_{\Gamma^{(k-1)}},\qquad k=1,\dots,M.
\]
By induction, $c_{\Gamma^{(M)}}=c_{\Gamma^{(0)}}$, i.e., $c_{\Gamma_1}=c_{\Gamma_2}$.

This completes the proof of Proposition~\ref{constantpro}.
\end{proof}
